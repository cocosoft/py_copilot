# 知识库系统优化集成方案

## 基于整理后的项目结构设计

---

## 一、现状分析

### 1.1 当前处理流程

```python
# document_processor.py 当前流程（6步）

Step 1: 文档解析
    ↓
Step 2: 文本清理
    ↓
Step 3: 智能分块
    ↓
Step 4: 实体识别与关系提取
    ↓
Step 5: 向量化处理
    ↓
Step 6: 知识图谱构建
```

### 1.2 缺失的环节

**问题：实体对齐未集成到处理流程**

当前流程在 Step 4（实体识别）后，直接将原始实体传递给 Step 6（知识图谱构建），**缺少实体对齐环节**。

这导致：
- 同一实体的不同表述无法统一
- 无法生成 KBEntity（知识库级实体）
- 无法建立跨文档的实体关联

### 3.2 批量处理API设计

**文件位置：** `app/api/v1/endpoints/knowledge_base/batch.py`

```python
from fastapi import APIRouter, Depends, HTTPException, BackgroundTasks
from sqlalchemy.orm import Session
from typing import List, Dict, Any, Optional
from pydantic import BaseModel

from app.core.database import get_db
from app.modules.auth.models import User
from app.modules.auth.dependencies import get_current_user
from app.tasks.knowledge.document_processing import process_document_task

router = APIRouter()


class BatchProcessRequest(BaseModel):
    """批量处理请求"""
    document_ids: List[int]
    options: Optional[Dict[str, Any]] = None


class BatchProcessResponse(BaseModel):
    """批量处理响应"""
    task_id: str
    status: str
    document_count: int
    message: str


@router.post("/knowledge-bases/{kb_id}/documents/batch-process", response_model=BatchProcessResponse)
async def batch_process_documents(
    kb_id: int,
    request: BatchProcessRequest,
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """
    批量处理知识库文档
    
    流程：
    1. 验证文档属于该知识库
    2. 创建批量任务组
    3. 异步执行处理
    
    Args:
        kb_id: 知识库ID
        request: 批量处理请求
        
    Returns:
        批量处理任务信息
    """
    try:
        # 验证知识库存在
        from app.modules.knowledge.models.knowledge_document import KnowledgeBase
        kb = db.query(KnowledgeBase).filter(KnowledgeBase.id == kb_id).first()
        if not kb:
            raise HTTPException(status_code=404, detail="知识库不存在")
        
        # 验证文档属于该知识库
        from app.modules.knowledge.models.knowledge_document import KnowledgeDocument
        valid_docs = db.query(KnowledgeDocument).filter(
            KnowledgeDocument.id.in_(request.document_ids),
            KnowledgeDocument.knowledge_base_id == kb_id
        ).all()
        
        valid_doc_ids = [doc.id for doc in valid_docs]
        invalid_count = len(request.document_ids) - len(valid_doc_ids)
        
        if invalid_count > 0:
            logger.warning(f"有 {invalid_count} 个文档不属于知识库 {kb_id}")
        
        if not valid_doc_ids:
            raise HTTPException(status_code=400, detail="没有有效的文档需要处理")
        
        # 创建批量任务
        from celery import group
        
        tasks = [
            process_document_task.s(doc_id, kb_id, request.options)
            for doc_id in valid_doc_ids
        ]
        
        # 提交任务组
        job = group(tasks).apply_async()
        
        return BatchProcessResponse(
            task_id=job.id,
            status="pending",
            document_count=len(valid_doc_ids),
            message=f"批量处理任务已创建，共 {len(valid_doc_ids)} 个文档"
        )
        
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"创建批量处理任务失败: {e}")
        raise HTTPException(status_code=500, detail=f"创建任务失败: {str(e)}")


@router.get("/knowledge-bases/{kb_id}/documents/batch-status/{task_id}")
async def get_batch_process_status(
    kb_id: int,
    task_id: str,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """
    获取批量处理任务状态
    
    Args:
        kb_id: 知识库ID
        task_id: 任务ID
        
    Returns:
        任务状态和进度
    """
    try:
        from celery.result import GroupResult
        
        result = GroupResult.restore(task_id)
        if not result:
            raise HTTPException(status_code=404, detail="任务不存在")
        
        # 计算进度
        total = len(result.results)
        completed = sum(1 for r in result.results if r.ready())
        successful = sum(1 for r in result.results if r.successful())
        
        return {
            "task_id": task_id,
            "status": "completed" if result.ready() else "processing",
            "progress": {
                "total": total,
                "completed": completed,
                "successful": successful,
                "failed": completed - successful,
                "percentage": int((completed / total) * 100) if total > 0 else 0
            }
        }
        
    except HTTPException:
        raise
    except Exception as e:
        logger.error(f"查询任务状态失败: {e}")
        raise HTTPException(status_code=500, detail=f"查询失败: {str(e)}")
```

### 3.3 Celery任务设计

**文件位置：** `app/tasks/knowledge/document_processing.py`

```python
from celery import shared_task
from typing import Dict, Any
import logging

logger = logging.getLogger(__name__)


@shared_task(bind=True, max_retries=3)
def process_document_task(self, document_id: int, knowledge_base_id: int, 
                          options: Dict[str, Any] = None) -> Dict[str, Any]:
    """
    处理单个文档的Celery任务
    
    Args:
        document_id: 文档ID
        knowledge_base_id: 知识库ID
        options: 处理选项
        
    Returns:
        处理结果
    """
    from app.core.database import SessionLocal
    from app.modules.knowledge.models.knowledge_document import KnowledgeDocument
    from app.services.knowledge.core.document_processor import DocumentProcessor
    
    db = SessionLocal()
    processor = DocumentProcessor()
    
    try:
        # 获取文档信息
        document = db.query(KnowledgeDocument).filter(
            KnowledgeDocument.id == document_id
        ).first()
        
        if not document:
            return {
                "success": False,
                "error": f"文档不存在: {document_id}"
            }
        
        # 执行处理
        result = processor.process_document(
            file_path=document.file_path,
            file_type=document.file_type,
            document_id=document_id,
            knowledge_base_id=knowledge_base_id,
            db=db
        )
        
        # 更新文档状态
        if result.get("success"):
            document.status = "processed"
            document.processing_result = result
        else:
            document.status = "failed"
            document.error_message = result.get("error")
        
        db.commit()
        
        return {
            "success": True,
            "document_id": document_id,
            "result": result
        }
        
    except Exception as e:
        logger.error(f"处理文档失败: {e}")
        db.rollback()
        
        # 重试逻辑
        if self.request.retries < self.max_retries:
            raise self.retry(countdown=60, exc=e)
        
        return {
            "success": False,
            "document_id": document_id,
            "error": str(e)
        }
    finally:
        db.close()
```
