class BaseTool(ABC):
    """工具基类 - 增强版"""
    
    def __init__(self):
        self._metadata = self._get_metadata()
        self._parameters = self._get_parameters()
        self._service_instances = {}  # 服务实例缓存
        self._execution_history = []   # 执行历史
    
    def inject_service(self, name: str, service_instance):
        """注入服务实例"""
        self._service_instances[name] = service_instance
    
    def get_service(self, name: str):
        """获取注入的服务"""
        return self._service_instances.get(name)
    
    def validate_parameters(self, params: dict) -> Tuple[bool, Optional[str]]:
        """
        统一参数验证
        
        验证规则：
        1. 必填参数检查
        2. 参数类型检查
        3. 枚举值检查
        4. 范围检查（数值）
        """
        for param in self._parameters:
            # 必填检查
            if param.required and param.name not in params:
                return False, f"缺少必填参数: {param.name}"
            
            # 类型检查
            if param.name in params:
                value = params[param.name]
                if not self._check_type(value, param.type):
                    return False, f"参数 {param.name} 类型错误，期望 {param.type}"
            
            # 枚举检查
            if param.enum and params.get(param.name) not in param.enum:
                return False, f"参数 {param.name} 值必须在 {param.enum} 中"
        
        return True, None
    
    def _check_type(self, value: Any, expected_type: str) -> bool:
        """检查参数类型"""
        type_map = {
            "string": str,
            "integer": int,
            "number": (int, float),
            "boolean": bool,
            "array": list,
            "object": dict
        }
        expected = type_map.get(expected_type)
        if expected:
            return isinstance(value, expected)
        return True
```

"""
工具服务注册中心

统一管理工具与服务的依赖关系，实现服务的延迟加载和单例管理
"""

from typing import Dict, Any, Type
from functools import lru_cache

class ToolServiceRegistry:
    """工具服务注册中心"""
    
    _instance = None
    _services: Dict[str, Any] = {}
    _service_classes: Dict[str, Type] = {}
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance
    
    def register_service_class(self, name: str, service_class: Type):
        """注册服务类（延迟实例化）"""
        self._service_classes[name] = service_class
    
    def get_service(self, name: str) -> Any:
        """获取服务实例（单例模式）"""
        if name not in self._services:
            if name in self._service_classes:
                service_class = self._service_classes[name]
                self._services[name] = service_class()
            else:
                raise KeyError(f"未注册的服务: {name}")
        return self._services[name]
    
    def inject_services_to_tool(self, tool: "BaseTool"):
        """向工具注入所需服务"""
        service_mapping = self._get_tool_service_mapping(tool)
        for service_name, attr_name in service_mapping.items():
            service = self.get_service(service_name)
            tool.inject_service(attr_name, service)
    
    def _get_tool_service_mapping(self, tool: "BaseTool") -> Dict[str, str]:
        """获取工具的服务依赖映射"""
        # 根据工具类型返回对应的服务映射
        mapping = {
            "web_search": {"web_search_service": "search_service"},
            "knowledge_search": {"knowledge_service": "knowledge_service"},
            "memory": {"memory_service": "memory_service"},
            "file_operation": {"file_storage_service": "file_service"},
            "image_processing": {"image_processing_service": "image_service"},
            "code_execution": {"command_execution_service": "command_service"},
        }
        return mapping.get(tool.metadata.name, {})

# 全局注册中心实例

"""
工具注册表 - 增强版

支持自动服务注入和工具发现
"""

from typing import Dict, Type, List
from app.modules.function_calling.base_tool import BaseTool
from app.modules.function_calling.tool_service_registry import tool_service_registry

class ToolRegistry:
    """工具注册表"""
    
    def __init__(self):
        self._tools: Dict[str, BaseTool] = {}
        self._tool_classes: Dict[str, Type[BaseTool]] = {}
    
    def register_tool(self, tool_class: Type[BaseTool]):
        """注册工具类"""
        # 创建临时实例获取元数据
        temp_instance = tool_class()
        tool_name = temp_instance.metadata.name
        
        self._tool_classes[tool_name] = tool_class
        
        # 创建正式实例并注入服务
        tool_instance = tool_class()
        tool_service_registry.inject_services_to_tool(tool_instance)
        
        self._tools[tool_name] = tool_instance
    
    def get_tool(self, name: str) -> BaseTool:
        """获取工具实例"""
        if name not in self._tools:
            raise KeyError(f"未注册的工具: {name}")
        return self._tools[name]
    
    def list_tools(self) -> List[Dict[str, Any]]:
        """列出所有工具"""
        return [
            {
                "name": tool.metadata.name,
                "display_name": tool.metadata.display_name,
                "description": tool.metadata.description,
                "category": tool.metadata.category,
                "version": tool.metadata.version,
                "icon": tool.metadata.icon,
                "tags": tool.metadata.tags,
                "is_active": getattr(tool.metadata, 'is_active', True)
            }
            for tool in self._tools.values()
        ]
    
    def auto_discover_tools(self):
        """自动发现并注册工具"""
        import importlib
        import pkgutil
        from app.modules.function_calling import tools
        
        for importer, modname, ispkg in pkgutil.iter_modules(tools.__path__):
            if modname.endswith('_tool'):
                module = importlib.import_module(f"app.modules.function_calling.tools.{modname}")
                for name, obj in module.__dict__.items():
                    if (isinstance(obj, type) and 
                        issubclass(obj, BaseTool) and 
                        obj is not BaseTool and
                        not getattr(obj, '_abstract', False)):
                        self.register_tool(obj)

# 全局注册表实例

                description="时间范围过滤",
                required=False,
                enum=["day", "week", "month", "year"]
            )
        ]
    
    def _get_service(self) -> WebSearchService:
        """获取搜索服务（延迟加载）"""

            self._search_service = WebSearchService()
        return self._search_service
    
    async def execute(self, **kwargs) -> ToolExecutionResult:
        """执行Web搜索"""
        import time
        start_time = time.time()
        
        try:
            # 1. 参数验证
            is_valid, error_msg = self.validate_parameters(kwargs)
            if not is_valid:
                return ToolExecutionResult(
                    success=False,
                    error=error_msg,
                    error_code="PARAM_INVALID",
                    execution_time=time.time() - start_time,
                    tool_name=self._metadata.name
                )

            query = kwargs.get("query")
            engine = kwargs.get("engine", "google")
            num_results = kwargs.get("num_results", 5)
            safe_search = kwargs.get("safe_search", True)
            time_range = kwargs.get("time_range")

            logger.info(f"[WebSearchTool] 执行搜索: query={query}, engine={engine}")
            service = self._get_service()
            
            search_result = service.search(
                query=query,
                engine=engine,
                safe_search=safe_search,
                num_results=num_results,
                time_range=time_range
            )
            
            # 4. 格式化结果
            formatted_result = self._format_search_result(search_result)
            
            execution_time = time.time() - start_time
            
            return ToolExecutionResult(
                success=True,
                result=formatted_result,
                execution_time=execution_time,
                tool_name=self._metadata.name,
                metadata={
                    "query": query,
                    "engine": engine,
                    "total_results": len(formatted_result.get("results", [])),
                    "search_time": execution_time
                }
            )
            
        except Exception as e:
            logger.error(f"[WebSearchTool] 搜索失败: {str(e)}", exc_info=True)
            return ToolExecutionResult(
                success=False,
                error=f"搜索失败: {str(e)}",
                error_code="SEARCH_ERROR",
                execution_time=time.time() - start_time,
                tool_name=self._metadata.name
            )
    
    def _format_search_result(self, raw_result: Dict[str, Any]) -> Dict[str, Any]:
        """格式化搜索结果"""
        if not raw_result.get("success"):
            raise Exception(raw_result.get("error", "搜索失败"))
        
        results = raw_result.get("results", [])
        
        # 格式化每个结果
        formatted_results = []
        for item in results:
            formatted_results.append({
                "title": item.get("title", ""),
                "url": item.get("url", ""),
                "snippet": item.get("snippet", ""),
                "source": item.get("source", ""),
                "timestamp": item.get("timestamp")
            })
        
        return {
            "results": formatted_results,
            "total_found": raw_result.get("total_results", len(formatted_results)),
            "query": raw_result.get("query", ""),
            "engine": raw_result.get("engine", "")
        }

    def _process_data(self, action: str, data: str, key: str, value: str, options: dict) -> dict:
        """根据action处理数据"""
        import base64
        from urllib.parse import quote, unquote
        
        # JSON处理
        if action == "parse_json":
            parsed = json.loads(data)
            return {
                "data": parsed,
                "type": type(parsed).__name__,
                "is_array": isinstance(parsed, list),
                "is_object": isinstance(parsed, dict)
            }
        
        elif action == "stringify_json":
            indent = options.get("indent", 2)
            obj = json.loads(data) if isinstance(data, str) else data
            return {"json": json.dumps(obj, indent=indent, ensure_ascii=False)}
        
        elif action == "validate_json":
            try:
                json.loads(data)
                return {"valid": True, "error": None}
            except json.JSONDecodeError as e:
                return {"valid": False, "error": str(e)}
        
        # 数组操作
        elif action == "sort_array":
            arr = json.loads(data) if isinstance(data, str) else data
            if not isinstance(arr, list):
                raise ValueError("数据必须是数组")
            reverse = options.get("reverse", False)
            if key:
                sorted_arr = sorted(arr, key=lambda x: x.get(key) if isinstance(x, dict) else x, reverse=reverse)
            else:
                sorted_arr = sorted(arr, reverse=reverse)
            return {"result": sorted_arr}
        
        elif action == "filter_array":
            arr = json.loads(data) if isinstance(data, str) else data
            if not isinstance(arr, list):
                raise ValueError("数据必须是数组")
            if key and value:
                filtered = [item for item in arr if isinstance(item, dict) and item.get(key) == value]
            else:
                filtered = arr
            return {"result": filtered, "count": len(filtered)}
        
        elif action == "unique_array":
            arr = json.loads(data) if isinstance(data, str) else data
            if not isinstance(arr, list):
                raise ValueError("数据必须是数组")
            seen = set()
            unique = []
            for item in arr:
                item_key = json.dumps(item, sort_keys=True) if isinstance(item, (dict, list)) else item
                if item_key not in seen:
                    seen.add(item_key)
                    unique.append(item)
            return {"result": unique, "original_count": len(arr), "unique_count": len(unique)}
        
        # 数据提取
        elif action == "get_keys":
            obj = json.loads(data) if isinstance(data, str) else data
            if isinstance(obj, dict):
                return {"keys": list(obj.keys())}
            else:
                raise ValueError("数据必须是对象")
        
        elif action == "get_values":
            obj = json.loads(data) if isinstance(data, str) else data
            if isinstance(obj, dict):
                return {"values": list(obj.values())}
            else:
                raise ValueError("数据必须是对象")
        
        # 编码转换
        elif action == "base64_encode":
            encoded = base64.b64encode(data.encode()).decode()
            return {"encoded": encoded}
        
        elif action == "base64_decode":
            decoded = base64.b64decode(data).decode()

                required=True
            ),
            ToolParameter(
                name="b",
                type="number",
                description="第二个操作数",
                required=False
            ),
            ToolParameter(
                name="precision",
                type="integer",
                description="结果精度（小数位数）",
                required=False,
                default=10
            ),
            ToolParameter(
                name="angle_unit",
                type="string",
                description="角度单位（用于三角函数）",
                required=False,
                default="radians",
                enum=["degrees", "radians"]
            )
        ]
    
    async def execute(self, **kwargs) -> ToolExecutionResult:
        """执行计算"""
        start_time = time.time()
        
        try:
            # 1. 参数验证
            is_valid, error_msg = self.validate_parameters(kwargs)
            if not is_valid:
                return ToolExecutionResult(
                    success=False,
                    error=error_msg,
                    error_code="PARAM_INVALID",
                    execution_time=time.time() - start_time,
                    tool_name=self._metadata.name
                )
            
            # 2. 提取参数
            operation = kwargs.get("operation")
            a = kwargs.get("a")
            b = kwargs.get("b")
            precision = kwargs.get("precision", 10)
            angle_unit = kwargs.get("angle_unit", "radians")
            
            # 3. 执行计算
            result = self._calculate(operation, a, b, angle_unit)
            
            # 4. 格式化结果
            if isinstance(result, float):
                result = round(result, precision)
            
            execution_time = time.time() - start_time
            
            return ToolExecutionResult(
                success=True,
                result={"value": result, "operation": operation},
                execution_time=execution_time,
                tool_name=self._metadata.name,
                metadata={"operation": operation, "precision": precision}
            )
            
        except ZeroDivisionError:
            return ToolExecutionResult(
                success=False,
                error="除数不能为零",
                error_code="DIVISION_BY_ZERO",
                execution_time=time.time() - start_time,
                tool_name=self._metadata.name
            )
        except Exception as e:
            logger.error(f"[CalculatorTool] 计算失败: {str(e)}", exc_info=True)
            return ToolExecutionResult(
                success=False,
                error=f"计算失败: {str(e)}",
                error_code="CALCULATION_ERROR",
                execution_time=time.time() - start_time,
                tool_name=self._metadata.name
            )
    
    def _calculate(self, operation: str, a, b, angle_unit: str):
        """执行具体计算"""
        a = float(a)
        if b is not None:
            b = float(b)
        
        # 基础运算
        if operation == "add":
            return a + b
        elif operation == "subtract":
            return a - b
        elif operation == "multiply":
            return a * b
        elif operation == "divide":
            return a / b
        elif operation == "modulo":
            return a % b
        
        # 科学计算
        elif operation == "power":
            return math.pow(a, b)
        elif operation == "sqrt":
            return math.sqrt(a)
        elif operation == "cbrt":
            return a ** (1/3)
        elif operation == "log":
            base = b if b is not None else 10
            return math.log(a, base)
        elif operation == "ln":
            return math.log(a)
        elif operation == "exp":
            return math.exp(a)
        elif operation == "abs":
            return abs(a)
        
        # 三角函数
        elif operation in ["sin", "cos", "tan", "asin", "acos", "atan"]:
            # 角度转弧度
            if angle_unit == "degrees":
                a = math.radians(a)
            
            if operation == "sin":
                return math.sin(a)
            elif operation == "cos":
                return math.cos(a)
            elif operation == "tan":
                return math.tan(a)
            elif operation == "asin":
                result = math.asin(a)
                return math.degrees(result) if angle_unit == "degrees" else result
            elif operation == "acos":
                result = math.acos(a)
                return math.degrees(result) if angle_unit == "degrees" else result
            elif operation == "atan":
                result = math.atan(a)
                return math.degrees(result) if angle_unit == "degrees" else result
        
        # 角度转换
        elif operation == "degrees":
            return math.degrees(a)
        elif operation == "radians":
            return math.radians(a)
        
        # 统计计算（单参数为数组）
        elif operation in ["sum", "mean", "median", "std", "min", "max"]:
            if isinstance(a, (list, tuple)):
                data = a
            else:
                raise ValueError(f"{operation}操作需要提供数组")
            
            if operation == "sum":
                return sum(data)
            elif operation == "mean":
                return sum(data) / len(data)
            elif operation == "median":
                sorted_data = sorted(data)
                n = len(sorted_data)
                if n % 2 == 0:
                    return (sorted_data[n//2-1] + sorted_data[n//2]) / 2
                else:
                    return sorted_data[n//2]
            elif operation == "std":
                mean = sum(data) / len(data)
                variance = sum((x - mean) ** 2 for x in data) / len(data)
                return math.sqrt(variance)
            elif operation == "min":
                return min(data)
            elif operation == "max":
                return max(data)
        
        else:
            raise ValueError(f"不支持的操作: {operation}")

                type="string",
                description="操作类型",
                required=True,
                enum=[
                    "now", "today", "timestamp",
                    "add", "diff", "between",
                    "format", "parse",
                    "is_weekend", "is_leap_year", "days_in_month"
                ]
            ),
            ToolParameter(
                name="datetime",
                type="string",
                description="日期时间字符串",
                required=False
            ),
            ToolParameter(
                name="format",
                type="string",
                description="格式字符串",
                required=False,
                default="%Y-%m-%d %H:%M:%S"
            ),
            ToolParameter(
                name="amount",
                type="integer",
                description="数量（用于add操作）",
                required=False
            ),
            ToolParameter(
                name="unit",
                type="string",
                description="单位",
                required=False,
                enum=["seconds", "minutes", "hours", "days", "weeks", "months", "years"]
            )
        ]
    
    async def execute(self, **kwargs) -> ToolExecutionResult:
        """执行时间操作"""
        import time
        start_time = time.time()
        
        try:
            action = kwargs.get("action")
            dt_str = kwargs.get("datetime")
            fmt = kwargs.get("format", "%Y-%m-%d %H:%M:%S")
            
            # 解析日期时间
            if dt_str:
                try:
                    dt = datetime.strptime(dt_str, fmt)
                except ValueError:
                    # 尝试自动解析
                    dt = datetime.fromisoformat(dt_str.replace('Z', '+00:00'))
            else:
                dt = datetime.now()
            
            # 执行操作
            if action == "now":
                result = {"datetime": datetime.now().isoformat()}
            elif action == "today":
                result = {"date": datetime.now().strftime("%Y-%m-%d")}
            elif action == "timestamp":
                result = {"timestamp": int(dt.timestamp())}
            elif action == "add":
                amount = kwargs.get("amount", 0)
                unit = kwargs.get("unit", "days")
                new_dt = self._add_time(dt, amount, unit)
                result = {"original": dt.isoformat(), "result": new_dt.isoformat()}
            elif action == "diff":
                dt2_str = kwargs.get("datetime2")
                if dt2_str:
                    dt2 = datetime.fromisoformat(dt2_str.replace('Z', '+00:00'))
                    diff = abs((dt2 - dt).total_seconds())
                    result = {"diff_seconds": diff, "diff_days": diff / 86400}
                else:
                    raise ValueError("diff操作需要提供第二个datetime参数")
            elif action == "format":
                result = {"formatted": dt.strftime(fmt)}
            elif action == "is_weekend":
                result = {"is_weekend": dt.weekday() >= 5}
            elif action == "is_leap_year":
                year = dt.year
                result = {"is_leap_year": year % 4 == 0 and (year % 100 != 0 or year % 400 == 0)}
            elif action == "days_in_month":
                import calendar
                result = {"days": calendar.monthrange(dt.year, dt.month)[1]}
            else:
                return ToolExecutionResult(
                    success=False,
                    error=f"不支持的操作: {action}",
                    error_code="UNSUPPORTED_ACTION",
                    execution_time=time.time() - start_time,
                    tool_name=self._metadata.name
                )
            
            return ToolExecutionResult(
                success=True,
                result=result,
                execution_time=time.time() - start_time,
                tool_name=self._metadata.name,
                metadata={"action": action}
            )
            
        except Exception as e:
            logger.error(f"[DateTimeTool] 操作失败: {str(e)}", exc_info=True)
            return ToolExecutionResult(
                success=False,
                error=f"时间日期操作失败: {str(e)}",
                error_code="DATETIME_ERROR",
                execution_time=time.time() - start_time,
                tool_name=self._metadata.name
            )
    
    def _add_time(self, dt: datetime, amount: int, unit: str) -> datetime:
        """添加时间"""
        if unit == "seconds":
            return dt + timedelta(seconds=amount)
        elif unit == "minutes":
            return dt + timedelta(minutes=amount)
        elif unit == "hours":
            return dt + timedelta(hours=amount)
        elif unit == "days":
            return dt + timedelta(days=amount)
        elif unit == "weeks":
            return dt + timedelta(weeks=amount)
        elif unit == "months":
            # 简单处理：加30天
            return dt + timedelta(days=amount * 30)
        elif unit == "years":
            # 简单处理：加365天
            return dt + timedelta(days=amount * 365)
        else:
            raise ValueError(f"不支持的时间单位: {unit}")
```

                ]
            ),
            ToolParameter(
                name="min_value",
                type="number",
                description="最小值（用于数字类型）",

            ),
            ToolParameter(
                name="max_value",
                type="number",
                description="最大值（用于数字类型）",
                required=False,
                default=100
            ),
            ToolParameter(
                name="length",
                type="integer",
                description="长度（用于字符串类型）",
                required=False,
                default=10
            ),
            ToolParameter(
                name="choices",
                type="array",
                description="选项列表（用于choice类型）",
                required=False
            ),
            ToolParameter(
                name="count",
                type="integer",
                description="生成数量",
                required=False,
                default=1
            )
        ]
    
    async def execute(self, **kwargs) -> ToolExecutionResult:
        """执行随机生成"""
        import time
        start_time = time.time()
        
        try:
            gen_type = kwargs.get("type")
            count = kwargs.get("count", 1)
            
            results = []
            for _ in range(count):
                if gen_type == "integer":
                    min_val = kwargs.get("min_value", 0)
                    max_val = kwargs.get("max_value", 100)
                    results.append(random.randint(int(min_val), int(max_val)))
                elif gen_type == "float":
                    min_val = kwargs.get("min_value", 0.0)
                    max_val = kwargs.get("max_value", 1.0)
                    results.append(random.uniform(min_val, max_val))
                elif gen_type == "boolean":
                    results.append(random.choice([True, False]))
                elif gen_type == "string":
                    length = kwargs.get("length", 10)
                    chars = string.ascii_letters + string.digits
                    results.append(''.join(random.choice(chars) for _ in range(length)))
                elif gen_type == "uuid":
                    results.append(str(uuid.uuid4()))
                elif gen_type == "password":
                    length = kwargs.get("length", 12)
                    chars = string.ascii_letters + string.digits + "!@#$%^&*"
                    results.append(''.join(random.choice(chars) for _ in range(length)))
                elif gen_type == "choice":
                    choices = kwargs.get("choices", [])
                    if choices:
                        results.append(random.choice(choices))
                    else:
                        raise ValueError("choice类型需要提供choices参数")
                elif gen_type == "shuffle":
                    choices = kwargs.get("choices", [])
                    if choices:
                        shuffled = choices.copy()
                        random.shuffle(shuffled)
                        results.append(shuffled)
                    else:
                        raise ValueError("shuffle类型需要提供choices参数")
                elif gen_type == "color":
                    results.append(f"#{random.randint(0, 0xFFFFFF):06x}")
                else:
                    raise ValueError(f"不支持的生成类型: {gen_type}")
            
            return ToolExecutionResult(
                success=True,
                result={"values": results if count > 1 else results[0]},
                execution_time=time.time() - start_time,
                tool_name=self._metadata.name,
                metadata={"type": gen_type, "count": count}
            )
            
        except Exception as e:
            logger.error(f"[RandomGeneratorTool] 生成失败: {str(e)}", exc_info=True)
            return ToolExecutionResult(
                success=False,
                error=f"随机生成失败: {str(e)}",
                error_code="GENERATION_ERROR",
                execution_time=time.time() - start_time,
                tool_name=self._metadata.name
            )

"""工具API - 更新版"""
from fastapi import APIRouter, HTTPException
from typing import Dict, Any

from app.modules.function_calling.tool_registry import tool_registry

router = APIRouter()

@router.get("/tools")
async def list_tools():
    """获取工具列表"""
    tools = tool_registry.list_tools()
    return {"success": True, "data": tools}

@router.get("/tools/{tool_name}")
async def get_tool(tool_name: str):
    """获取工具详情"""
    try:
        tool = tool_registry.get_tool(tool_name)
        return {
            "success": True,
            "data": {
                "name": tool.metadata.name,
                "display_name": tool.metadata.display_name,
                "description": tool.metadata.description,
                "category": tool.metadata.category,
                "version": tool.metadata.version,
                "icon": tool.metadata.icon,
                "tags": tool.metadata.tags,
                "parameters": [
                    {
                        "name": p.name,
                        "type": p.type,
                        "description": p.description,
                        "required": p.required,
                        "default": p.default,
                        "enum": p.enum
                    }
                    for p in tool.parameters
                ]
            }
        }
    except KeyError:
        raise HTTPException(status_code=404, detail=f"工具不存在: {tool_name}")

@router.post("/tools/{tool_name}/execute")
async def execute_tool(tool_name: str, params: Dict[str, Any]):
    """执行工具"""
    try:
        tool = tool_registry.get_tool(tool_name)
        result = await tool.execute(**params)
        return {
            "success": result.success,
            "data": {
                "result": result.result,
                "execution_time": result.execution_time
            },
            "error": result.error,
            "error_code": result.error_code
        }
    except KeyError:
        raise HTTPException(status_code=404, detail=f"工具不存在: {tool_name}")
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.get("/tools/categories")
async def list_categories():
    """获取工具分类"""
    tools = tool_registry.list_tools()
    categories = list(set(tool["category"] for tool in tools))
    return {"success": True, "data": categories}
