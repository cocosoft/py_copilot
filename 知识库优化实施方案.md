def format_results(self, results: Dict[str, Any], source: str = "vector") -> List[Dict[str, Any]]:
    """统一格式化检索结果"""
    formatted = []
    
    if not results or not results.get('ids') or not results['ids'][0]:
        return formatted
    
    for i, doc_id in enumerate(results['ids'][0]):
        metadata = results['metadatas'][0][i] if results.get('metadatas') else {}
        
        formatted.append({
            'id': doc_id,
            'title': metadata.get('title', '无标题'),
            'content': results['documents'][0][i] if results.get('documents') else '',
            'score': 1 - results['distances'][0][i] if 'distances' in results and results['distances'][0] else 0.0,
            'knowledge_base_id': metadata.get('knowledge_base_id'),
            'file_type': metadata.get('file_type'),
            'created_at': metadata.get('created_at'),
            'source': source,
            'metadata': metadata
        })
    
    formatted.sort(key=lambda x: x['score'], reverse=True)
    return formatted

```python
"""重排序服务模块"""
import logging
from typing import List, Dict, Any, Optional
from sentence_transformers import CrossEncoder

logger = logging.getLogger(__name__)


class RerankService:
    """重排序服务，使用专业的CrossEncoder模型"""
    
    def __init__(self, model_name: str = "BAAI/bge-reranker-large"):
        """
        初始化重排序服务
        
        Args:
            model_name: 重排序模型名称，默认使用BAAI/bge-reranker-large
        """
        self.model_name = model_name
        self.model = None
        self.available = False
        
        try:
            self.model = CrossEncoder(model_name, max_length=512)
            self.available = True
            logger.info(f"重排序模型加载成功: {model_name}")
        except Exception as e:
            logger.error(f"重排序模型加载失败: {e}")
            self.available = False
    
    def rerank(self, query: str, documents: List[Dict[str, Any]], 
               top_k: int = 10) -> List[Dict[str, Any]]:
        """
        对检索结果进行重排序
        
        Args:
            query: 查询文本
            documents: 候选文档列表
            top_k: 返回前k个结果
            
        Returns:
            重排序后的文档列表
        """
        if not self.available or not documents:
            logger.warning("重排序模型不可用，返回原始排序结果")
            return sorted(documents, key=lambda x: x.get('score', 0), reverse=True)[:top_k]
        
        try:
            # 准备输入对
            pairs = [[query, doc.get('content', '')] for doc in documents]
            
            # 计算重排序分数
            scores = self.model.predict(pairs)
            
            # 更新文档分数
            for i, doc in enumerate(documents):
                doc['original_score'] = doc.get('score', 0)
                doc['rerank_score'] = float(scores[i])
                # 综合分数：原始分数 * 0.3 + 重排序分数 * 0.7
                doc['score'] = doc['original_score'] * 0.3 + doc['rerank_score'] * 0.7
            
            # 按新分数排序
            reranked = sorted(documents, key=lambda x: x['score'], reverse=True)
            
            logger.info(f"重排序完成: 输入 {len(documents)} 条，输出 {len(reranked[:top_k])} 条")
            return reranked[:top_k]
            
        except Exception as e:
            logger.error(f"重排序失败: {e}")
            return sorted(documents, key=lambda x: x.get('score', 0), reverse=True)[:top_k]
```

# 修改点5: 添加向量化重试机制
async def process_document_async(self, document_id: int, knowledge_base_id: int):
    """异步处理文档 - 添加重试机制"""
    from app.core.database import SessionLocal
    import asyncio
    
    db = SessionLocal()
    max_retries = 3
    retry_count = 0
    
    try:
        while retry_count < max_retries:
            try:
                logger.info(f"开始处理文档: {document_id}, 尝试 {retry_count + 1}/{max_retries}")
                
                # 获取文档
                document = db.query(KnowledgeDocument).filter(KnowledgeDocument.id == document_id).first()
                if not document:
                    logger.error(f"文档不存在: {document_id}")
                    return
                
                # 更新处理状态
                document.document_metadata["processing_status"] = "processing"
                document.document_metadata["retry_count"] = retry_count
                db.commit()
                
                # 处理文档
                from app.services.knowledge.document_processor import DocumentProcessor
                document_processor = DocumentProcessor()
                
                processing_result = document_processor.process_document(
                    document.file_path,
                    document.file_type,
                    document.id,
                    document.knowledge_base_id,
                    db
                )
                
                if processing_result.get("success"):
                    logger.info(f"文档处理成功: {document_id}")
                    document.is_vectorized = True
                    document.document_metadata["processing_status"] = "completed"
                    document.document_metadata["vectorization_rate"] = processing_result.get("vectorization_rate", 0)
                    db.commit()
                    break
                else:
                    raise Exception(processing_result.get("error", "处理失败"))
                    
            except Exception as e:
                retry_count += 1
                if retry_count >= max_retries:
                    raise
                
                wait_time = 2 ** retry_count  # 指数退避
                logger.warning(f"处理失败，{wait_time}秒后重试: {e}")
                await asyncio.sleep(wait_time)
        
    except Exception as e:
        logger.error(f"文档处理最终失败: {document_id} - {e}")
        try:
            document = db.query(KnowledgeDocument).filter(KnowledgeDocument.id == document_id).first()
            if document:
                document.document_metadata["processing_status"] = "failed"
                document.document_metadata["processing_error"] = str(e)
                document.document_metadata["retry_count"] = retry_count
                db.commit()
        except:
            pass
    finally:
        db.close()
```

# 修改点6: 集成spaCy NER模型
import spacy
from typing import List, Dict, Any, Tuple, Optional
import logging

logger = logging.getLogger(__name__)


class AdvancedTextProcessor:
    """高级文本处理模块，集成专业NER模型"""
    
    def __init__(self, config_file: str = "entity_config.json"):
        """
        初始化高级文本处理器
        
        Args:
            config_file: 实体配置文件路径
        """
        from app.services.knowledge.entity_config_manager import EntityConfigManager
        self.config_manager = EntityConfigManager(config_file)
        
        # 延迟初始化
        self.use_jieba = False
        self.jieba_initialized = False
        self.nlp = None
        self.nlp_initialized = False
        
        logger.info("高级文本处理器初始化成功")
    
    def _initialize_nlp(self):
        """延迟初始化spaCy NER模型"""
        if not self.nlp_initialized:
            try:
                self.nlp = spacy.load("zh_core_web_sm")
                self.nlp_initialized = True
                logger.info("spaCy NER模型加载成功")
            except Exception as e:
                logger.warning(f"spaCy模型加载失败: {e}，将使用基于规则的实体识别")
                self.nlp_initialized = True
    
    def extract_entities_relationships(self, text: str) -> Tuple[List[Dict[str, Any]], List[Dict[str, Any]]]:
        """
        提取实体和关系 - 使用NER模型+规则混合策略
        
        Args:
            text: 输入文本
            
        Returns:
            (实体列表, 关系列表)
        """
        self._initialize_nlp()
        
        # 1. 使用spaCy NER模型提取实体
        ner_entities = []
        if self.nlp:
            doc = self.nlp(text)
            for ent in doc.ents:
                ner_entities.append({
                    "text": ent.text,
                    "type": ent.label_,
                    "start_pos": ent.start_char,
                    "end_pos": ent.end_char,
                    "confidence": 0.9,
                    "source": "ner_model"
                })
        
        # 2. 使用规则提取补充实体
        rule_entities = self._rule_based_entity_extraction(text)
        
        # 3. 合并实体（去重）
        merged_entities = self._merge_entities(ner_entities, rule_entities)
        
        # 4. 提取关系
        relationships = self._extract_relationships(text, merged_entities)
        
        return merged_entities, relationships
    
    def _merge_entities(self, ner_entities: List[Dict], rule_entities: List[Dict]) -> List[Dict]:
        """
        合并NER模型和规则提取的实体，去重
        
        Args:
            ner_entities: NER模型提取的实体
            rule_entities: 规则提取的实体
            
        Returns:
            合并后的实体列表
        """
        merged = []
        seen_positions = set()
        
        # 优先使用NER模型的结果
        for entity in ner_entities:
            pos_key = (entity["start_pos"], entity["end_pos"])
            if pos_key not in seen_positions:
                merged.append(entity)
                seen_positions.add(pos_key)
        
        # 添加规则提取的实体（不重复）
        for entity in rule_entities:
            pos_key = (entity["start_pos"], entity["end_pos"])
            if pos_key not in seen_positions:
                # 检查是否与已有实体重叠
                overlap = False
                for existing in merged:
                    if not (entity["end_pos"] <= existing["start_pos"] or 
                           entity["start_pos"] >= existing["end_pos"]):
                        overlap = True
                        break
                
                if not overlap:
                    merged.append(entity)
                    seen_positions.add(pos_key)
        
        return merged
```

**回滚触发条件：**
- 向量化失败率>5%
- 搜索功能不可用
- 系统崩溃或严重性能下降

**回滚步骤：**
```bash
# 1. 停止服务
systemctl stop knowledge-base

# 2. 恢复代码

git push origin main

# 3. 恢复数据库（如需要）

# 4. 恢复向量数据库
cp -r backend/chroma_db_backup_YYYYMMDD backend/chroma_db

# 5. 重启服务
systemctl start knowledge-base
```
