#### 2.1.3 优化文档分块策略

**修改文件**: `advanced_text_processor.py`

```python
def semantic_chunking(self, text: str, file_type: str = "txt", max_chunk_size: int = None, 
                     min_chunk_size: int = None, overlap: int = None) -> List[str]:
    """智能语义分块 - 根据文件类型自适应参数"""
    
    # 根据文件类型设置默认参数
    type_configs = {
        ".pdf": {"max": 800, "min": 200, "overlap": 100},
        ".docx": {"max": 1000, "min": 200, "overlap": 100},
        ".txt": {"max": 1200, "min": 300, "overlap": 150},
        ".md": {"max": 1000, "min": 200, "overlap": 100},
    }
    
    config = type_configs.get(file_type, {})
    max_chunk_size = max_chunk_size or config.get("max", 1000)
    min_chunk_size = min_chunk_size or config.get("min", 200)
    overlap = overlap or config.get("overlap", 100)
    
    # 根据内容类型选择分块策略
    if self._is_structured_content(text):
        return self._structured_chunking(text, max_chunk_size, min_chunk_size, overlap)
    else:
        return self._semantic_chunking_impl(text, max_chunk_size, min_chunk_size, overlap)

def _is_structured_content(self, text: str) -> bool:
    """判断是否为结构化内容（如Markdown、代码等）"""
    # 检测Markdown标题
    if re.search(r'^#{1,6}\s', text, re.MULTILINE):
        return True
    # 检测代码块
    if '```' in text:
        return True
    return False

def _structured_chunking(self, text: str, max_chunk_size: int, 
                        min_chunk_size: int, overlap: int) -> List[str]:
    """基于文档结构的智能分块"""
    chunks = []
    
    # 按Markdown标题分割
    sections = re.split(r'(?=^#{1,6}\s)', text, flags=re.MULTILINE)
    
    current_chunk = ""
    for section in sections:
        if not section.strip():
            continue
            
        # 如果当前块加上新章节不超过限制，则合并
        if len(current_chunk) + len(section) <= max_chunk_size:
            current_chunk += section
        else:
            # 保存当前块
            if len(current_chunk) >= min_chunk_size:
                chunks.append(current_chunk.strip())
            
            # 如果单个章节超过限制，需要进一步分割
            if len(section) > max_chunk_size:
                sub_chunks = self._split_large_section(section, max_chunk_size, overlap)
                chunks.extend(sub_chunks)
                current_chunk = ""
            else:
                current_chunk = section
    
    # 添加最后一个块
    if current_chunk and len(current_chunk) >= min_chunk_size:
        chunks.append(current_chunk.strip())
    
    return chunks
```

import logging
from typing import List, Dict, Any, Optional
from sentence_transformers import CrossEncoder

logger = logging.getLogger(__name__)

class RerankService:
    """重排序服务，使用专业的CrossEncoder模型"""
    
    def __init__(self, model_name: str = "BAAI/bge-reranker-large"):
        self.model_name = model_name
        self.model = None
        self.available = False
        
        try:
            self.model = CrossEncoder(model_name, max_length=512)
            self.available = True
            logger.info(f"重排序模型加载成功: {model_name}")
        except Exception as e:
            logger.error(f"重排序模型加载失败: {e}")
            self.available = False
    
    def rerank(self, query: str, documents: List[Dict[str, Any]], 
               top_k: int = 10) -> List[Dict[str, Any]]:
        """
        对检索结果进行重排序
        
        Args:
            query: 查询文本
            documents: 候选文档列表
            top_k: 返回前k个结果
            
        Returns:
            重排序后的文档列表
        """
        if not self.available or not documents:
            # 如果模型不可用，按原始分数排序
            return sorted(documents, key=lambda x: x.get('score', 0), reverse=True)[:top_k]
        
        try:
            # 准备输入对
            pairs = [[query, doc.get('content', '')] for doc in documents]
            
            # 计算重排序分数
            scores = self.model.predict(pairs)
            
            # 更新文档分数
            for i, doc in enumerate(documents):
                doc['original_score'] = doc.get('score', 0)
                doc['rerank_score'] = float(scores[i])
                # 综合分数：原始分数 * 0.3 + 重排序分数 * 0.7
                doc['score'] = doc['original_score'] * 0.3 + doc['rerank_score'] * 0.7
            
            # 按新分数排序
            reranked = sorted(documents, key=lambda x: x['score'], reverse=True)
            
            logger.info(f"重排序完成: 输入 {len(documents)} 条，输出 {len(reranked[:top_k])} 条")
            return reranked[:top_k]
            
        except Exception as e:
            logger.error(f"重排序失败: {e}")
            return sorted(documents, key=lambda x: x.get('score', 0), reverse=True)[:top_k]
```

def format_results(self, results: Dict[str, Any], source: str = "vector") -> List[Dict[str, Any]]:
    """统一格式化检索结果"""
    formatted = []
    
    if not results or not results.get('ids') or not results['ids'][0]:
        return formatted
    
    for i, doc_id in enumerate(results['ids'][0]):
        metadata = results['metadatas'][0][i] if results.get('metadatas') else {}
        
        formatted.append({
            'id': doc_id,
            'title': metadata.get('title', '无标题'),
            'content': results['documents'][0][i] if results.get('documents') else '',
            'score': 1 - results['distances'][0][i] if 'distances' in results and results['distances'][0] else 0.0,
            'knowledge_base_id': metadata.get('knowledge_base_id'),
            'file_type': metadata.get('file_type'),
            'created_at': metadata.get('created_at'),
            'source': source,  # 标记来源
            'metadata': metadata
        })
    
    # 按相似度排序
    formatted.sort(key=lambda x: x['score'], reverse=True)
    return formatted

```python
import spacy
from typing import List, Dict, Any, Tuple, Optional
import logging

logger = logging.getLogger(__name__)

class AdvancedTextProcessor:
    """高级文本处理模块，集成专业NER模型"""
    
    def __init__(self, config_file: str = "entity_config.json"):
        # 初始化配置管理器
        self.config_manager = EntityConfigManager(config_file)
        
        # 延迟初始化jieba中文分词器
        self.use_jieba = False
        self.jieba_initialized = False
        
        # 延迟加载spaCy NER模型
        self.nlp = None
        self.nlp_initialized = False
        
        logger.info("高级文本处理器初始化成功")
    
    def _initialize_nlp(self):
        """延迟初始化spaCy NER模型"""
        if not self.nlp_initialized:
            try:
                # 加载中文NER模型
                self.nlp = spacy.load("zh_core_web_sm")
                self.nlp_initialized = True
                logger.info("spaCy NER模型加载成功")
            except Exception as e:
                logger.warning(f"spaCy模型加载失败: {e}，将使用基于规则的实体识别")
                self.nlp_initialized = True
    
    def extract_entities_relationships(self, text: str) -> Tuple[List[Dict[str, Any]], List[Dict[str, Any]]]:
        """提取实体和关系 - 使用NER模型+规则混合策略"""
        self._initialize_nlp()
        
        # 1. 使用spaCy NER模型提取实体
        ner_entities = []
        if self.nlp:
            doc = self.nlp(text)
            for ent in doc.ents:
                ner_entities.append({
                    "text": ent.text,
                    "type": ent.label_,
                    "start_pos": ent.start_char,
                    "end_pos": ent.end_char,
                    "confidence": 0.9,  # NER模型置信度较高
                    "source": "ner_model"
                })
        
        # 2. 使用规则提取补充实体
        rule_entities = self._rule_based_entity_extraction(text)
        
        # 3. 合并实体（去重）
        merged_entities = self._merge_entities(ner_entities, rule_entities)
        
        # 4. 提取关系
        relationships = self._extract_relationships(text, merged_entities)
        
        return merged_entities, relationships
    
    def _merge_entities(self, ner_entities: List[Dict], rule_entities: List[Dict]) -> List[Dict]:
        """合并NER模型和规则提取的实体，去重"""
        merged = []
        seen_positions = set()
        
        # 优先使用NER模型的结果
        for entity in ner_entities:
            pos_key = (entity["start_pos"], entity["end_pos"])
            if pos_key not in seen_positions:
                merged.append(entity)
                seen_positions.add(pos_key)
        
        # 添加规则提取的实体（不重复）
        for entity in rule_entities:
            pos_key = (entity["start_pos"], entity["end_pos"])
            if pos_key not in seen_positions:
                # 检查是否与已有实体重叠
                overlap = False
                for existing in merged:
                    if not (entity["end_pos"] <= existing["start_pos"] or 
                           entity["start_pos"] >= existing["end_pos"]):
                        overlap = True
                        break
                
                if not overlap:
                    merged.append(entity)
                    seen_positions.add(pos_key)
        
        return merged
```

            self.available = False
    
    def create_entity(self, entity_id: str, entity_text: str, 
                     entity_type: str, document_id: int,
                     confidence: float = 0.7) -> bool:
        """创建实体节点"""
        if not self.available:
            return False
        
        try:
            with self.driver.session() as session:
                session.run("""
                    MERGE (e:Entity {id: $entity_id})
                    SET e.text = $entity_text,
                        e.type = $entity_type,
                        e.document_id = $document_id,
                        e.confidence = $confidence,
                        e.created_at = datetime()
                """, entity_id=entity_id, entity_text=entity_text,
                   entity_type=entity_type, document_id=document_id,
                   confidence=confidence)
            return True
        except Exception as e:
            logger.error(f"创建实体失败: {e}")
            return False
    
    def create_relationship(self, source_id: str, target_id: str,
                           relationship_type: str, 
                           confidence: float = 0.7) -> bool:
        """创建关系边"""
        if not self.available:
            return False
        
        try:
            with self.driver.session() as session:
                session.run("""
                    MATCH (source:Entity {id: $source_id})
                    MATCH (target:Entity {id: $target_id})
                    MERGE (source)-[r:RELATES {type: $rel_type}]->(target)
                    SET r.confidence = $confidence,
                        r.created_at = datetime()
                """, source_id=source_id, target_id=target_id,
                   rel_type=relationship_type, confidence=confidence)
            return True
        except Exception as e:
            logger.error(f"创建关系失败: {e}")
            return False
    
    def find_entity_paths(self, source_text: str, target_text: str, 
                         max_depth: int = 3) -> List[List[Dict]]:
        """查找两个实体之间的路径"""
        if not self.available:
            return []
        
        try:
            with self.driver.session() as session:
                result = session.run("""
                    MATCH path = (source:Entity)-[*1..$max_depth]-(target:Entity)
                    WHERE source.text = $source_text 
                      AND target.text = $target_text
                    RETURN path
                    LIMIT 10
                """, source_text=source_text, target_text=target_text,
                   max_depth=max_depth)
                
                paths = []
                for record in result:
                    path = record["path"]
                    path_data = []
                    for node in path.nodes:
                        path_data.append({
                            "id": node["id"],
                            "text": node["text"],
                            "type": node["type"]
                        })
                    paths.append(path_data)
                
                return paths
        except Exception as e:
            logger.error(f"查找路径失败: {e}")
            return []
    
    def get_entity_neighbors(self, entity_text: str, 
                            limit: int = 10) -> List[Dict]:
        """获取实体的邻居节点"""
        if not self.available:
            return []
        
        try:
            with self.driver.session() as session:
                result = session.run("""
                    MATCH (e:Entity {text: $entity_text})-[r]-(neighbor)
                    RETURN neighbor, r
                    LIMIT $limit
                """, entity_text=entity_text, limit=limit)
                
                neighbors = []
                for record in result:
                    node = record["neighbor"]
                    rel = record["r"]
                    neighbors.append({
                        "id": node["id"],
                        "text": node["text"],
                        "type": node["type"],
                        "relationship": rel["type"]
                    })
                
                return neighbors
        except Exception as e:
            logger.error(f"获取邻居节点失败: {e}")
            return []
    
    def close(self):
        """关闭连接"""
        if self.driver:
            self.driver.close()
```

#### 2.4.3 优化实体链接算法

**修改文件**: `graph_builder.py`

```python
def _entity_linking(self, entities: List[DocumentEntity]) -> List[Dict]:
    """实体消歧和跨文档链接 - 改进版本"""
    if not entities:
        return []
    
    # 按实体类型分组
    entities_by_type = {}
    for entity in entities:
        if entity.entity_type not in entities_by_type:
            entities_by_type[entity.entity_type] = []
        entities_by_type[entity.entity_type].append(entity)
    
    linked_entities = []
    cluster_id = 0
    
    # 对每种实体类型进行聚类
    for entity_type, type_entities in entities_by_type.items():
        if len(type_entities) <= 1:
            linked_entities.append({
                "cluster_id": cluster_id,
                "entity_type": entity_type,
                "representative_text": type_entities[0].entity_text,
                "entities": type_entities,
                "confidence": type_entities[0].confidence
            })
            cluster_id += 1
            continue
        
        # 使用改进的聚类算法
        clusters = self._cluster_entities_advanced(type_entities)
        
        for cluster in clusters:
            if cluster:
                # 选择置信度最高的实体作为代表
                representative = max(cluster, key=lambda e: e.confidence)
                linked_entities.append({
                    "cluster_id": cluster_id,
                    "entity_type": entity_type,
                    "representative_text": representative.entity_text,
                    "entities": cluster,
                    "confidence": np.mean([e.confidence for e in cluster])
                })
                cluster_id += 1
    
    return linked_entities

def _cluster_entities_advanced(self, entities: List[DocumentEntity], 
                               similarity_threshold: float = 0.8) -> List[List[DocumentEntity]]:
    """改进的实体聚类算法"""
    if len(entities) <= 1:
        return [entities]
    
    # 1. 精确匹配聚类（文本完全相同）
    exact_match_clusters = {}
    for entity in entities:
        key = entity.entity_text.lower().strip()
        if key not in exact_match_clusters:
            exact_match_clusters[key] = []
        exact_match_clusters[key].append(entity)
    
    # 2. 对未聚类的实体使用相似度聚类
    remaining_entities = []
    clusters = list(exact_match_clusters.values())
    
    for cluster in clusters:
        if len(cluster) == 1:
            remaining_entities.append(cluster[0])
        else:
            # 已经是聚类好的
            pass
    
    # 3. 使用文本相似度对剩余实体进行聚类
    if remaining_entities:
        texts = [e.entity_text for e in remaining_entities]
        
        # 使用多种相似度指标
        vectorizer = TfidfVectorizer(analyzer='char', ngram_range=(2, 4))
        try:
            tfidf_matrix = vectorizer.fit_transform(texts)
            similarity_matrix = cosine_similarity(tfidf_matrix)
            
            # 使用层次聚类
            from sklearn.cluster import AgglomerativeClustering
            clustering = AgglomerativeClustering(
                n_clusters=None,
                distance_threshold=1 - similarity_threshold,
                linkage='average'
            )
            labels = clustering.fit_predict(1 - similarity_matrix)
            
            # 组织聚类结果
            similarity_clusters = {}
            for idx, label in enumerate(labels):
                if label not in similarity_clusters:
                    similarity_clusters[label] = []
                similarity_clusters[label].append(remaining_entities[idx])
            
            clusters.extend(similarity_clusters.values())
            
        except Exception as e:
            logger.warning(f"相似度聚类失败: {e}")
            # 每个实体单独成组
            for entity in remaining_entities:
                clusters.append([entity])
    
    return clusters
```
