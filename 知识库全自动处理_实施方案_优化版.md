```python
# backend/app/services/knowledge/entity_resolution_service.py

"""
实体融合服务 - 基于现有 LLMEntityDisambiguator 实现
"""

from typing import Dict, List, Optional, Set, Tuple
from dataclasses import dataclass
from collections import defaultdict
import numpy as np
from difflib import SequenceMatcher
import logging

from sqlalchemy.orm import Session
from app.services.knowledge.llm_extractor import LLMEntityDisambiguator
from app.models.unified_knowledge import UnifiedKnowledgeUnit

logger = logging.getLogger(__name__)


@dataclass
class EntityCluster:
    """实体簇"""
    cluster_id: str
    canonical_name: str
    entity_type: str
    members: List[Dict]
    confidence: float


class EntityResolutionService:
    """
    实体融合服务
    
    基于已有的 LLMEntityDisambiguator 实现实体对齐和聚类
    """
    
    def __init__(self, db: Session, knowledge_base_id: int):
        self.db = db
        self.knowledge_base_id = knowledge_base_id
        self.disambiguator = LLMEntityDisambiguator(db)
        self.similarity_threshold = 0.85
        
    def resolve_entities(self, entities: List[Dict], 
                        context: str = None) -> List[EntityCluster]:
        """
        执行实体融合
        
        Args:
            entities: 待融合实体列表
            context: 上下文文本
            
        Returns:
            实体簇列表
        """
        logger.info(f"开始实体融合: {len(entities)} 个实体")
        
        # 步骤1: 消歧（复用已有实现）
        disambiguated = self.disambiguator.disambiguate_entities(
            entities, context
        )
        
        # 步骤2: 聚类
        clusters = self._cluster_entities(disambiguated)
        
        # 步骤3: 与已有实体对齐
        aligned_clusters = self._align_with_existing(clusters)
        
        logger.info(f"实体融合完成: {len(clusters)} 个簇")
        
        return aligned_clusters
    
    def _cluster_entities(self, entities: List[Dict]) -> List[EntityCluster]:
        """实体聚类"""
        if not entities:
            return []
        
        # 按类型分组聚类
        type_groups = defaultdict(list)
        for e in entities:
            type_groups[e.get('type', 'UNKNOWN')].append(e)
        
        clusters = []
        cluster_id = 0
        
        for entity_type, group in type_groups.items():
            # 简单聚类：同名实体归为一类
            name_groups = defaultdict(list)
            for e in group:
                name_groups[e.get('text', '')].append(e)
            
            for name, members in name_groups.items():
                cluster = EntityCluster(
                    cluster_id=f"cluster_{cluster_id}",
                    canonical_name=name,
                    entity_type=entity_type,
                    members=members,
                    confidence=0.9 if len(members) > 1 else 0.7
                )
                clusters.append(cluster)
                cluster_id += 1
        
        return clusters
    
    def _align_with_existing(self, clusters: List[EntityCluster]) -> List[EntityCluster]:
        """与已有实体对齐"""
        # 查询知识库中已有实体
        existing_units = self.db.query(UnifiedKnowledgeUnit).filter(
            UnifiedKnowledgeUnit.knowledge_base_id == self.knowledge_base_id,
            UnifiedKnowledgeUnit.unit_type == 'ENTITY'
        ).all()
        
        existing_entities = [
            {
                'id': u.unit_id,
                'name': u.title,
                'type': u.metadata.get('entity_type') if u.metadata else None
            }
            for u in existing_units
        ]
        
        # 对齐逻辑
        for cluster in clusters:
            for existing in existing_entities:
                similarity = self._calculate_similarity(
                    cluster.canonical_name,
                    existing.get('name', '')
                )
                if similarity >= self.similarity_threshold:
                    cluster.canonical_name = existing.get('name')
                    cluster.confidence = max(cluster.confidence, similarity)
                    break
        
        return clusters
    
    def _calculate_similarity(self, name1: str, name2: str) -> float:
        """计算名称相似度"""
        return SequenceMatcher(None, name1.lower(), name2.lower()).ratio()
```

```python
# backend/app/api/v1/endpoints/knowledge_graph_config.py

from fastapi import APIRouter, Depends, HTTPException, BackgroundTasks
from pydantic import BaseModel
from typing import List, Optional
from sqlalchemy.orm import Session

from app.core.database import get_db
from app.core.security import get_current_user
from app.models.user import User

router = APIRouter()


class BatchProcessRequest(BaseModel):
    """批量处理请求"""
    document_ids: List[int]
    options: Optional[dict] = None


class BatchProcessResponse(BaseModel):
    """批量处理响应"""
    task_id: str
    status: str
    document_count: int
    message: str


@router.post("/knowledge-bases/{kb_id}/documents/batch-process", 
             response_model=BatchProcessResponse)
async def batch_process_documents(
    kb_id: int,
    request: BatchProcessRequest,
    background_tasks: BackgroundTasks,
    db: Session = Depends(get_db),
    current_user: User = Depends(get_current_user)
):
    """
    批量处理文档
    
    触发文档的完整处理流程：
    1. 实体提取
    2. 实体融合
    3. 向量化
    4. 知识图谱构建
    """
    from celery import group
    from app.tasks.entity_extraction_tasks import extract_entities_task
    
    try:
        # 创建批量任务
        tasks = [
            extract_entities_task.s(str(doc_id), str(kb_id), request.options)
            for doc_id in request.document_ids
        ]
        
        # 并行执行
        job = group(tasks).apply_async()
        
        return BatchProcessResponse(
            task_id=job.id,
            status="pending",
            document_count=len(request.document_ids),
            message="批量处理任务已创建"
        )
        
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"创建任务失败: {str(e)}")


@router.get("/tasks/{task_id}/status")
async def get_task_status(
    task_id: str,
    current_user: User = Depends(get_current_user)
):
    """获取任务状态"""
    from app.tasks.knowledge_graph_tasks import get_task_status
    
    status = get_task_status(task_id)
    
    if not status:
        raise HTTPException(status_code=404, detail="任务不存在")
    
    return status
```

### 6.3 Celery任务增强

```python
# backend/app/tasks/entity_alignment_tasks.py

"""
实体融合对齐任务
"""

from celery import shared_task
from typing import Dict, List, Any
from datetime import datetime
import logging

from app.core.database import SessionLocal
from app.services.knowledge.entity_resolution_service import EntityResolutionService

logger = logging.getLogger(__name__)


@shared_task(bind=True, max_retries=3)
def align_entities_task(self, document_id: str, knowledge_base_id: str) -> Dict[str, Any]:
    """
    实体融合对齐任务
    
    Args:
        document_id: 文档ID
        knowledge_base_id: 知识库ID
        
    Returns:
        融合结果
    """
    task_id = self.request.id
    
    try:
        logger.info(f"开始实体融合对齐: doc={document_id}")
        
        db = SessionLocal()
        
        try:
            # 获取文档实体
            from app.models.unified_knowledge import UnifiedKnowledgeUnit
            
            entity_units = db.query(UnifiedKnowledgeUnit).filter(
                UnifiedKnowledgeUnit.document_id == document_id,
                UnifiedKnowledgeUnit.unit_type == 'ENTITY'
            ).all()
            
            entities = [
                {
                    'text': u.title,
                    'type': u.metadata.get('entity_type') if u.metadata else 'UNKNOWN',
                    'unit_id': u.unit_id
                }
                for u in entity_units
            ]
            
            # 执行融合
            resolution_service = EntityResolutionService(db, int(knowledge_base_id))
            clusters = resolution_service.resolve_entities(entities)
            
            # 保存融合结果
            for cluster in clusters:
                # 更新实体的规范名称
                for member in cluster.members:
                    unit = db.query(UnifiedKnowledgeUnit).filter(
                        UnifiedKnowledgeUnit.unit_id == member.get('unit_id')
                    ).first()
                    
                    if unit:
                        unit.metadata = unit.metadata or {}
                        unit.metadata['canonical_name'] = cluster.canonical_name
                        unit.metadata['cluster_id'] = cluster.cluster_id
            
            db.commit()
            
            result = {
                'success': True,
                'document_id': document_id,
                'cluster_count': len(clusters),
                'entity_count': len(entities),
                'completed_at': datetime.now().isoformat()
            }
            
            logger.info(f"实体融合对齐完成: {document_id}, 簇数={len(clusters)}")
            
            return result
            
        finally:
            db.close()
            
    except Exception as e:
        logger.error(f"实体融合对齐失败: {e}")
        
        if self.request.retries < self.max_retries:
            raise self.retry(countdown=60, exc=e)
        
        return {
            'success': False,
            'error': str(e),
            'document_id': document_id
        }
```
