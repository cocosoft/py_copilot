```
KnowledgeBase (1)
    ├── UnifiedKnowledgeUnit (N) - 统一知识单元
    │       ├── DOCUMENT - 文档
    │       ├── CHUNK - 片段
    │       ├── ENTITY - 实体
    │       └── CONCEPT - 概念
    │
    ├── KnowledgeDocument (N) - 知识文档
    │       └── DocumentChunk (N) - 文档片段
    │
    └── KnowledgeUnitAssociation (N) - 单元关联

DocumentEntity (N) - 文档级实体
    └── 关联到 → KBEntity → GlobalEntity (分层实体)
```

```python
# 完整的文档处理流水线（在现有任务基础上扩展）

from celery import chain, group, chord
from app.tasks.entity_extraction_tasks import extract_entities_task
from app.tasks.knowledge_graph_tasks import (
    build_document_layer_task,
    build_knowledge_base_layer_task, 
    build_global_layer_task
)
from app.tasks.entity_alignment_tasks import align_entities_task

# ========== 完整处理流水线 ==========

def create_document_processing_pipeline(document_id: str, knowledge_base_id: str):
    """
    创建文档完整处理流水线
    
    流程:
    1. 实体提取
    2. 实体融合对齐
    3. 文档层构建
    4. 知识库层构建
    5. 全局层构建
    6. 向量化
    7. 质量评估
    """
    
    pipeline = chain(
        # 步骤1: 实体提取
        extract_entities_task.s(document_id, knowledge_base_id),
        
        # 步骤2: 实体融合对齐（依赖实体提取结果）
        align_entities_task.s(knowledge_base_id),
        
        # 步骤3: 构建文档层图谱
        build_document_layer_task.s(knowledge_base_id),
        
        # 步骤4: 构建知识库层图谱
        build_knowledge_base_layer_task.s(knowledge_base_id),
        
        # 步骤5: 构建全局层图谱
        build_global_layer_task.s(knowledge_base_id),
        
        # 步骤6: 向量化处理
        vectorize_document_task.s(document_id, knowledge_base_id),
        
        # 步骤7: 质量评估
        assess_quality_task.s(document_id, knowledge_base_id),
    )
    
    return pipeline


# ========== 批量处理流水线 ==========

def create_batch_processing_pipeline(document_ids: List[str], knowledge_base_id: str):
    """
    批量文档处理流水线
    
    使用 chord 实现并行处理后汇总
    """
    
    # 并行处理每个文档
    document_tasks = group(
        create_document_processing_pipeline(doc_id, knowledge_base_id)
        for doc_id in document_ids
    )
    
    # 所有文档处理完成后，执行知识库层汇总
    pipeline = chord(
        document_tasks,
        finalize_knowledge_base_task.s(knowledge_base_id)
    )
    
    return pipeline
```

```python
# backend/app/utils/entity_recognizer.py

"""
命名实体识别模块 - 基于LLM的实体提取

替换现有空实现，提供完整的NER能力
"""

from typing import Dict, List, Optional, Set, Tuple, Union
import json
import logging
from dataclasses import dataclass

from app.modules.llm.services.llm_service import LLMService

logger = logging.getLogger(__name__)


@dataclass
class Entity:
    """实体数据结构"""
    name: str
    entity_type: str
    start_pos: int
    end_pos: int
    confidence: float
    context: str
    attributes: Optional[Dict] = None


@dataclass
class Relation:
    """关系数据结构"""
    source: str
    target: str
    relation_type: str
    confidence: float
    evidence: str


class LLMEntityRecognizer:
    """
    基于LLM的实体识别器
    
    使用大语言模型进行命名实体识别和关系抽取
    """
    
    # 实体类型定义
    ENTITY_TYPES = {
        "PERSON": "人物",
        "ORGANIZATION": "组织",
        "LOCATION": "地点",
        "TIME": "时间",
        "EVENT": "事件",
        "PRODUCT": "产品",
        "TECHNOLOGY": "技术",
        "CONCEPT": "概念"
    }
    
    # 关系类型定义
    RELATION_TYPES = {
        "WORKS_FOR": "就职于",
        "LOCATED_IN": "位于",
        "FOUNDED_BY": "由...创立",
        "PART_OF": "属于",
        "USES": "使用",
        "RELATED_TO": "相关于"
    }
    
    def __init__(self, model_provider: str = "openai"):
        """
        初始化实体识别器
        
        Args:
            model_provider: 模型提供商 (openai/azure/local)
        """
        self.llm_service = LLMService()
        self.model_provider = model_provider
        self.is_available = True
        self.available = True
        
    def recognize_entities(self, text: str) -> List[Dict[str, Union[str, float, int]]]:
        """
        识别文本中的命名实体
        
        Args:
            text: 输入文本
            
        Returns:
            实体列表
        """
        try:
            # 构建提示词
            prompt = self._build_ner_prompt(text)
            
            # 调用LLM
            response = self.llm_service.generate(
                prompt=prompt,
                temperature=0.1,
                max_tokens=2000
            )
            
            # 解析结果
            entities = self._parse_ner_response(response, text)
            
            logger.info(f"实体识别完成: 发现 {len(entities)} 个实体")
            return entities
            
        except Exception as e:
            logger.error(f"实体识别失败: {e}")
            return []
    
    def extract_relations(self, text: str, entities: List[Dict]) -> List[Dict]:
        """
        抽取实体间关系
        
        Args:
            text: 输入文本
            entities: 已识别的实体列表
            
        Returns:
            关系列表
        """
        try:
            # 构建提示词
            prompt = self._build_relation_prompt(text, entities)
            
            # 调用LLM
            response = self.llm_service.generate(
                prompt=prompt,
                temperature=0.1,
                max_tokens=2000
            )
            
            # 解析结果
            relations = self._parse_relation_response(response)
            
            logger.info(f"关系抽取完成: 发现 {len(relations)} 个关系")
            return relations
            
        except Exception as e:
            logger.error(f"关系抽取失败: {e}")
            return []
    
    def _build_ner_prompt(self, text: str) -> str:
        """构建NER提示词"""
        entity_types_desc = "\n".join([f"- {k}: {v}" for k, v in self.ENTITY_TYPES.items()])
        
        prompt = f"""请从以下文本中提取命名实体。

实体类型:
{entity_types_desc}

要求:
1. 识别所有符合上述类型的实体
2. 返回JSON格式: [{{"name": "实体名", "type": "实体类型", "start_pos": 开始位置, "end_pos": 结束位置, "confidence": 置信度0-1}}]
3. 确保位置信息准确
4. 置信度根据识别确定性给出

文本:
{text}

输出:"""
        return prompt
    
    def _build_relation_prompt(self, text: str, entities: List[Dict]) -> str:
        """构建关系抽取提示词"""
        entity_list = "\n".join([f"- {e['name']} ({e['type']})" for e in entities])
        relation_types_desc = "\n".join([f"- {k}: {v}" for k, v in self.RELATION_TYPES.items()])
        
        prompt = f"""请从以下文本中抽取实体间的关系。

已知实体:
{entity_list}

关系类型:
{relation_types_desc}

要求:
1. 只抽取上述实体之间的关系
2. 返回JSON格式: [{{"source": "源实体名", "target": "目标实体名", "relation_type": "关系类型", "confidence": 置信度0-1, "evidence": "证据文本"}}]
3. 证据文本为关系中出现的原文片段

文本:
{text}

输出:"""
        return prompt
    
    def _parse_ner_response(self, response: str, original_text: str) -> List[Dict]:
        """解析NER响应"""
        try:
            # 提取JSON部分
            json_str = self._extract_json(response)
            entities = json.loads(json_str)
            
            # 验证和清理
            validated = []
            for e in entities:
                if self._validate_entity(e, original_text):
                    validated.append(e)
            
            return validated
        except Exception as e:
            logger.error(f"解析NER响应失败: {e}")
            return []
    
    def _parse_relation_response(self, response: str) -> List[Dict]:
        """解析关系抽取响应"""
        try:
            json_str = self._extract_json(response)
            relations = json.loads(json_str)
            return relations if isinstance(relations, list) else []
        except Exception as e:
            logger.error(f"解析关系响应失败: {e}")
            return []
    
    def _extract_json(self, text: str) -> str:
        """从文本中提取JSON"""
        # 查找方括号包裹的内容
        start = text.find('[')
        end = text.rfind(']')
        if start != -1 and end != -1:
            return text[start:end+1]
        return text
    
    def _validate_entity(self, entity: Dict, text: str) -> bool:
        """验证实体有效性"""
        required_fields = ['name', 'type', 'start_pos', 'end_pos']
        if not all(f in entity for f in required_fields):
            return False
        
        # 验证类型
        if entity['type'] not in self.ENTITY_TYPES:
            return False
        
        # 验证位置
        if entity['start_pos'] < 0 or entity['end_pos'] > len(text):
            return False
        
        return True


# 保留原有接口兼容
class EntityRecognizer(LLMEntityRecognizer):
    """向后兼容的实体识别器"""
    pass
```

### 3.2 实体融合模块（新增）

```python
# backend/app/utils/entity_resolution.py

"""
实体融合模块 - 实体消歧、共指消解、实体对齐

基于现有 UnifiedKnowledgeUnit 模型进行实体融合
"""

from typing import Dict, List, Optional, Set, Tuple
from dataclasses import dataclass
from collections import defaultdict
import numpy as np
from difflib import SequenceMatcher
import logging

from app.models.unified_knowledge import UnifiedKnowledgeUnit, KnowledgeUnitAssociation
from app.modules.llm.services.llm_service import LLMService

logger = logging.getLogger(__name__)


@dataclass
class EntityCluster:
    """实体簇 - 融合后的实体组"""
    cluster_id: str
    canonical_entity: Dict  # 规范实体
    members: List[Dict]     # 簇内成员
    confidence: float       # 融合置信度
    fusion_strategy: str    # 融合策略


class EntityResolutionEngine:
    """
    实体融合引擎
    
    实现实体消歧、共指消解和实体对齐
    """
    
    def __init__(self, similarity_threshold: float = 0.85):
        """
        初始化融合引擎
        
        Args:
            similarity_threshold: 相似度阈值
        """
        self.similarity_threshold = similarity_threshold
        self.llm_service = LLMService()
        
    def resolve_entities(self, entities: List[Dict], 
                        context: str,
                        existing_entities: Optional[List[Dict]] = None) -> List[EntityCluster]:
        """
        执行完整的实体融合流程
        
        Args:
            entities: 待融合的新实体列表
            context: 上下文文本
            existing_entities: 已有实体库（用于对齐）
            
        Returns:
            实体簇列表
        """
        logger.info(f"开始实体融合: 新实体={len(entities)}, 已有实体={len(existing_entities) if existing_entities else 0}")
        
        # 步骤1: 实体消歧
        disambiguated = self._disambiguate_entities(entities, context)
        logger.info(f"实体消歧完成: {len(disambiguated)} 个实体")
        
        # 步骤2: 共指消解
        coreference_resolved = self._resolve_coreference(disambiguated, context)
        logger.info(f"共指消解完成: {len(coreference_resolved)} 个实体")
        
        # 步骤3: 实体聚类
        clusters = self._cluster_entities(coreference_resolved)
        logger.info(f"实体聚类完成: {len(clusters)} 个簇")
        
        # 步骤4: 与已有实体对齐
        if existing_entities:
            clusters = self._align_with_existing(clusters, existing_entities)
            logger.info(f"实体对齐完成")
        
        return clusters
    
    def _disambiguate_entities(self, entities: List[Dict], context: str) -> List[Dict]:
        """
        实体消歧 - 区分同名不同义的实体
        
        例如: "苹果" → 水果(Apple Inc.) vs 公司(Apple Fruit)
        """
        # 按名称分组
        name_groups = defaultdict(list)
        for e in entities:
            name_groups[e['name']].append(e)
        
        disambiguated = []
        
        for name, group in name_groups.items():
            if len(group) == 1:
                # 唯一名称，无需消歧
                disambiguated.append(group[0])
            else:
                # 同名实体，需要消歧
                for entity in group:
                    # 使用上下文进行消歧
                    entity_context = self._extract_context(entity, context)
                    entity['disambiguation_context'] = entity_context
                    entity['disambiguated_name'] = f"{name}({entity['type']})"
                    disambiguated.append(entity)
        
        return disambiguated
    
    def _resolve_coreference(self, entities: List[Dict], context: str) -> List[Dict]:
        """
        共指消解 - 识别不同表述指向同一实体
        
        例如: "乔布斯" = "Steve Jobs" = "他" = "Apple创始人"
        """
        # 使用LLM进行共指消解
        try:
            prompt = self._build_coreference_prompt(entities, context)
            response = self.llm_service.generate(prompt=prompt, temperature=0.1)
            
            # 解析共指链
            coreference_chains = self._parse_coreference_response(response)
            
            # 合并共指实体
            resolved = self._merge_coreferences(entities, coreference_chains)
            return resolved
            
        except Exception as e:
            logger.error(f"共指消解失败: {e}")
            return entities
    
    def _cluster_entities(self, entities: List[Dict]) -> List[EntityCluster]:
        """
        实体聚类 - 基于相似度将实体分组
        """
        if not entities:
            return []
        
        # 按类型分组
        type_groups = defaultdict(list)
        for e in entities:
            type_groups[e['type']].append(e)
        
        clusters = []
        cluster_id = 0
        
        for entity_type, group in type_groups.items():
            # 计算相似度矩阵
            n = len(group)
            similarity_matrix = np.zeros((n, n))
            
            for i in range(n):
                for j in range(i+1, n):
                    sim = self._calculate_similarity(group[i], group[j])
                    similarity_matrix[i][j] = sim
                    similarity_matrix[j][i] = sim
            
            # 基于相似度聚类
            visited = set()
            for i in range(n):
                if i in visited:
                    continue
                
                # 找到相似实体
                cluster_members = [group[i]]
                visited.add(i)
                
                for j in range(i+1, n):
                    if similarity_matrix[i][j] >= self.similarity_threshold:
                        cluster_members.append(group[j])
                        visited.add(j)
                
                # 创建实体簇
                canonical = self._select_canonical_entity(cluster_members)
                cluster = EntityCluster(
                    cluster_id=f"cluster_{cluster_id}",
                    canonical_entity=canonical,
                    members=cluster_members,
                    confidence=np.mean([similarity_matrix[i][j] 
                                      for j in range(n) 
                                      if j != i and similarity_matrix[i][j] > 0]),
                    fusion_strategy="similarity_clustering"
                )
                clusters.append(cluster)
                cluster_id += 1
        
        return clusters
    
    def _align_with_existing(self, clusters: List[EntityCluster], 
                            existing: List[Dict]) -> List[EntityCluster]:
        """
        与已有实体对齐
        """
        aligned_clusters = []
        
        for cluster in clusters:
            best_match = None
            best_score = 0
            
            for existing_entity in existing:
                score = self._calculate_similarity(
                    cluster.canonical_entity, 
                    existing_entity
                )
                if score > best_score and score >= self.similarity_threshold:
                    best_score = score
                    best_match = existing_entity
            
            if best_match:
                # 对齐到已有实体
                cluster.canonical_entity['aligned_to'] = best_match.get('id')
                cluster.canonical_entity['alignment_confidence'] = best_score
            
            aligned_clusters.append(cluster)
        
        return aligned_clusters
    
    def _calculate_similarity(self, entity1: Dict, entity2: Dict) -> float:
        """
        计算实体相似度
        
        综合以下维度:
        - 名称相似度
        - 类型一致性
        - 上下文相似度
        """
        # 名称相似度
        name_sim = SequenceMatcher(
            None, 
            entity1.get('name', '').lower(), 
            entity2.get('name', '').lower()
        ).ratio()
        
        # 类型一致性
        type_match = 1.0 if entity1.get('type') == entity2.get('type') else 0.0
        
        # 综合相似度
        similarity = 0.6 * name_sim + 0.4 * type_match
        
        return similarity
    
    def _select_canonical_entity(self, members: List[Dict]) -> Dict:
        """选择规范实体（选择最完整、最准确的代表）"""
        # 选择名称最长的（通常更完整）
        canonical = max(members, key=lambda x: len(x.get('name', '')))
        
        # 添加别名
        aliases = set(m['name'] for m in members if m['name'] != canonical['name'])
        canonical['aliases'] = list(aliases)
        
        return canonical
    
    def _extract_context(self, entity: Dict, text: str, window: int = 50) -> str:
        """提取实体上下文"""
        start = max(0, entity['start_pos'] - window)
        end = min(len(text), entity['end_pos'] + window)
        return text[start:end]
    
    def _build_coreference_prompt(self, entities: List[Dict], context: str) -> str:
        """构建共指消解提示词"""
        entity_list = "\n".join([f"- {e['name']} ({e['type']})" for e in entities])
        
        prompt = f"""请识别以下实体列表中的共指关系（不同表述指向同一实体）。

实体列表:
{entity_list}

上下文:
{context[:1000]}

要求:
1. 识别所有指向同一实体的不同表述
2. 返回JSON格式: [["实体1", "实体2", "实体3"], ["实体4", "实体5"]]
3. 每个子列表是一组共指实体
4. 只包含确定是同一实体的表述

输出:"""
        return prompt
    
    def _parse_coreference_response(self, response: str) -> List[List[str]]:
        """解析共指响应"""
        try:
            import json
            # 提取JSON
            start = response.find('[')
            end = response.rfind(']')
            if start != -1 and end != -1:
                return json.loads(response[start:end+1])
            return []
        except:
            return []
    
    def _merge_coreferences(self, entities: List[Dict], 
                           chains: List[List[str]]) -> List[Dict]:
        """合并共指实体"""
        merged = entities.copy()
        
        for chain in chains:
            # 找到链中所有实体
            chain_entities = [e for e in merged if e['name'] in chain]
            
            if len(chain_entities) > 1:
                # 保留第一个作为主实体
                primary = chain_entities[0]
                primary['coreference_chain'] = chain
                
                # 移除其他共指实体
                for e in chain_entities[1:]:
                    if e in merged:
                        merged.remove(e)
        
        return merged

entity_resolution_engine = EntityResolutionEngine()
```

```python
"""

from celery import shared_task

@shared_task(bind=True, max_retries=3)
def vectorize_document_task(self, document_id: str, knowledge_base_id: str) -> Dict[str, Any]:
    """
    文档向量化任务
    
    将文档片段转换为向量并存储到 ChromaDB
    
    Args:
        document_id: 文档ID
        knowledge_base_id: 知识库ID
        
    Returns:
        向量化结果
    """
    task_id = self.request.id
    
    try:
        logger.info(f"开始文档向量化: doc={document_id}")
        
        # 更新任务状态
        from app.tasks.knowledge_graph_tasks import update_task_status
        update_task_status(task_id, 'started', 0, '开始向量化')
        
        db = SessionLocal()
        
        try:
            # 获取文档的所有知识单元（CHUNK类型）
            units = db.query(UnifiedKnowledgeUnit).filter(
                UnifiedKnowledgeUnit.document_id == document_id,
                UnifiedKnowledgeUnit.unit_type == 'CHUNK',
                UnifiedKnowledgeUnit.status == 'active'
            ).all()
            
            update_task_status(task_id, 'progress', 20, f'获取到 {len(units)} 个片段')
            
            # 初始化向量化服务
            from sentence_transformers import SentenceTransformer
            import chromadb
            
            # 加载模型
            model = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')
            
            # 连接ChromaDB
            chroma_client = chromadb.HttpClient(host='localhost', port=8000)
            collection = chroma_client.get_or_create_collection(
                name=f"kb_{knowledge_base_id}",
                metadata={"knowledge_base_id": knowledge_base_id}
            )
            
            update_task_status(task_id, 'progress', 40, '模型加载完成')
            
            # 批量向量化
            vectorized_count = 0
            batch_size = 32
            
            for i in range(0, len(units), batch_size):
                batch = units[i:i+batch_size]
                
                # 准备文本（可加入实体标签增强）
                texts = []
                metadatas = []
                ids = []
                
                for unit in batch:
                    # 获取关联的实体信息
                    entity_tags = _get_entity_tags(unit, db)
                    
                    # 构建增强文本
                    enhanced_text = unit.content
                    if entity_tags:
                        enhanced_text = f"[实体: {', '.join(entity_tags)}] {unit.content}"
                    
                    texts.append(enhanced_text)
                    metadatas.append({
                        'unit_id': unit.unit_id,
                        'document_id': document_id,
                        'knowledge_base_id': knowledge_base_id,
                        'title': unit.title or '',
                        'entity_tags': entity_tags
                    })
                    ids.append(unit.unit_id)
                
                # 生成向量
                embeddings = model.encode(texts).tolist()
                
                # 存储到ChromaDB
                collection.add(
                    embeddings=embeddings,
                    documents=texts,
                    metadatas=metadatas,
                    ids=ids
                )
                
                # 更新单元状态
                for unit in batch:
                    unit.vector_status = 'completed'
                    unit.vectorized_at = datetime.now()
                    unit.vector_id = unit.unit_id
                
                vectorized_count += len(batch)
                progress = 40 + int((vectorized_count / len(units)) * 50)
                update_task_status(task_id, 'progress', progress, 
                                 f'已向量化 {vectorized_count}/{len(units)} 个片段')
            
            db.commit()
            
            result = {
                'success': True,
                'document_id': document_id,
                'knowledge_base_id': knowledge_base_id,
                'vectorized_count': vectorized_count,
                'completed_at': datetime.now().isoformat()
            }
            
            update_task_status(task_id, 'success', 100, '向量化完成', result)
            logger.info(f"文档向量化完成: {document_id}, 共 {vectorized_count} 个片段")
            
            return result
            
        finally:
            db.close()
            
    except Exception as e:
        logger.error(f"向量化失败: {e}")
        
        if self.request.retries < self.max_retries:
            raise self.retry(countdown=60, exc=e)
        
        return {
            'success': False,
            'error': str(e),
            'document_id': document_id
        }
```

```python
            update_task_status(task_id, 'progress', 50, '评估向量化质量...')
            vectorization_quality = _assess_vectorization_quality(units)
            
            update_task_status(task_id, 'progress', 70, '评估图谱构建质量...')
            graph_quality = _assess_graph_quality(document_id, knowledge_base_id)
            
            update_task_status(task_id, 'progress', 90, '计算综合评分...')
            
            # 计算综合质量分
            overall_score = (
                extraction_quality['score'] * 0.4 +
                vectorization_quality['score'] * 0.3 +
                graph_quality['score'] * 0.3
            )
            
            # 更新知识单元质量分
            for unit in units:
                unit.quality_score = overall_score
                unit.quality_dimensions = {
                    'extraction': extraction_quality,
                    'vectorization': vectorization_quality,
                    'graph': graph_quality
                }
            
            db.commit()
            
            result = {
                'success': True,
                'document_id': document_id,
                'overall_score': overall_score,
                'dimensions': {
                    'extraction': extraction_quality,
                    'vectorization': vectorization_quality,
                    'graph': graph_quality
                },
                'assessed_at': datetime.now().isoformat()
            }

            'document_id': document_id
        }


def _assess_extraction_quality(units: List[UnifiedKnowledgeUnit]) -> Dict[str, Any]:
    """评估实体提取质量"""
    entity_units = [u for u in units if u.unit_type == 'ENTITY']
    chunk_units = [u for u in units if u.unit_type == 'CHUNK']
    
    if not chunk_units:
        return {'score': 0, 'reason': '无文档片段'}
    
    # 计算实体密度
    entity_density = len(entity_units) / len(chunk_units)
    
    # 计算平均置信度
    avg_confidence = 0.85  # 假设值，实际应从实体记录中获取
    
    # 综合评分
    score = min(100, (entity_density * 20 + avg_confidence * 80))
    
    return {
        'score': score,
        'entity_count': len(entity_units),
        'chunk_count': len(chunk_units),
        'entity_density': entity_density,
        'avg_confidence': avg_confidence
    }


def _assess_vectorization_quality(units: List[UnifiedKnowledgeUnit]) -> Dict[str, Any]:
    """评估向量化质量"""
    chunk_units = [u for u in units if u.unit_type == 'CHUNK']
    
    if not chunk_units:
        return {'score': 0, 'reason': '无文档片段'}
    
    # 计算向量化成功率
    vectorized = [u for u in chunk_units if u.vector_status == 'completed']
    success_rate = len(vectorized) / len(chunk_units)
    
    # 评分
    score = success_rate * 100
    
    return {
        'score': score,
        'total_chunks': len(chunk_units),
        'vectorized_chunks': len(vectorized),
        'success_rate': success_rate
    }
```
