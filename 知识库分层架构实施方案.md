```python
def extract_document_entities_parallel(self, document_id: int, max_workers: int = 4) -> Dict[str, int]:
    """并行提取文档中所有片段的实体"""
    # 获取文档的所有片段
    chunks = self.db.query(DocumentChunk).filter(
        DocumentChunk.document_id == document_id
    ).all()
    
    # 使用线程池并行处理
    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = []
        for chunk in chunks:
            future = executor.submit(
                self._extract_chunk_entities,
                chunk.id
            )
            futures.append(future)
        
        # 收集结果
        completed = 0
        failed = 0
        for future in as_completed(futures):
            try:
                future.result()
                completed += 1
            except Exception as e:
                failed += 1
                logger.error(f"处理片段失败: {e}")
    
    return {
        "total_chunks": len(chunks),
        "completed": completed,
        "failed": failed
    }
```

def validate_preconditions(self, document_id: int, target_stage: str) -> Dict[str, Any]:
    """验证处理阶段的前置条件"""
    # 定义前置条件映射
    preconditions = {
        DocumentProcessingStatus.TEXT_EXTRACTED.value: [
            DocumentProcessingStatus.IDLE.value,
            DocumentProcessingStatus.FAILED.value
        ],
        DocumentProcessingStatus.CHUNKED.value: [
            DocumentProcessingStatus.TEXT_EXTRACTED.value,
            DocumentProcessingStatus.CHUNKING_FAILED.value
        ],
        # 其他前置条件...
    }
    
    # 检查前置条件
    required_statuses = preconditions.get(target_stage, [])
    if current_status not in required_statuses:
        return {
            "valid": False,
            "message": f"当前状态 '{current_status}' 不满足阶段 '{target_stage}' 的前置条件",
            "required_statuses": required_statuses
        }
    
    return {"valid": True, "message": "前置条件满足"}

## 9. 风险评估

### 9.1 潜在风险

| 风险 | 影响 | 缓解措施 |
|------|------|----------|
| 内存泄漏 | 系统崩溃 | 主动内存释放、监控内存使用 |
| 处理超时 | 用户体验差 | 设置合理的超时时间、异步处理 |
| 数据库性能 | 处理速度慢 | 优化数据库查询、使用索引 |
| 并行处理冲突 | 数据不一致 | 合理的锁机制、事务管理 |

### 9.2 应对策略

- **内存管理：** 主动释放内存，监控内存使用
- **超时处理：** 异步处理，设置合理的超时时间
- **数据库优化：** 优化查询，使用索引，批量操作
- **并行控制：** 合理的线程池大小，避免过度并行
