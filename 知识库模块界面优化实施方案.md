## 一、项目概述

### 1.1 项目背景

知识库模块作为 Py Copilot IV 的核心功能，目前存在架构冗余、界面复杂、性能瓶颈等问题。为提升用户体验和系统可维护性，需要进行全面优化。

### 1.2 优化范围

- **前端架构重构**: 状态管理、路由结构、组件组织
- **界面体验优化**: 布局设计、视觉统一、交互改进
- **性能优化**: 虚拟列表、代码分割、资源加载

### 1.3 预期成果

- 操作效率提升 50%
- 首屏加载时间减少 50%
- 支持万级数据流畅处理
- 代码可维护性提升 30%

  setDocumentSort: (sort: SortConfig) => void;
  uploadDocument: (file: File, kbId: string) => Promise<void>;
  deleteDocument: (docId: string) => Promise<void>;
  downloadDocument: (docId: string) => Promise<void>;
  
  // 批量操作
  batchVectorize: (docIds: string[]) => Promise<void>;
  batchDelete: (docIds: string[]) => Promise<void>;

  setActiveTab: (tab: KnowledgeState['activeTab']) => void;
  toggleSidebar: () => void;
  toggleDetailPanel: () => void;
  
  // 搜索操作
  setSearchQuery: (query: string) => void;
  executeSearch: () => Promise<void>;
  addSearchHistory: (query: string) => void;
  clearSearchHistory: () => void;
}

// Store 创建
export const useKnowledgeStore = create<KnowledgeState & KnowledgeActions>()(
  devtools(
    immer(
      persist(
        (set, get) => ({
          // 初始状态
          knowledgeBases: [],
          currentKnowledgeBase: null,
          knowledgeBaseLoading: false,
          knowledgeBaseError: null,
          
          documents: [],
          selectedDocuments: [],
          documentFilters: {},
          documentSort: { field: 'createdAt', order: 'desc' },
          documentPagination: { page: 1, pageSize: 20, total: 0 },
          documentLoading: false,
          documentError: null,
          
          viewMode: 'list',
          activeTab: 'documents',
          sidebarCollapsed: false,
          detailPanelVisible: true,
          
          isProcessing: false,
          processingProgress: 0,
          processingQueue: [],
          
          searchQuery: '',
          searchFilters: {},
          searchHistory: [],
          searchSuggestions: [],
          
          // Actions 实现
          fetchKnowledgeBases: async () => {
            set({ knowledgeBaseLoading: true, knowledgeBaseError: null });
            try {
              const response = await knowledgeApi.getKnowledgeBases();
              set({ knowledgeBases: response.data });
            } catch (error) {
              set({ knowledgeBaseError: error.message });
            } finally {
              set({ knowledgeBaseLoading: false });
            }
          },
          
          setCurrentKnowledgeBase: (kb) => {
            set({ currentKnowledgeBase: kb });
            if (kb) {
              get().fetchDocuments(kb.id);
            }
          },
          
          // ... 其他 actions
        }),
        {
          name: 'knowledge-store',
          partialize: (state) => ({
            viewMode: state.viewMode,
            sidebarCollapsed: state.sidebarCollapsed,
            searchHistory: state.searchHistory
          })
        }
      )
    ),
    { name: 'KnowledgeStore' }
  )
);

  });

  describe('文档操作', () => {
    it('应该能够选择多个文档', () => {
      const { result } = renderHook(() => useKnowledgeStore());
      const docs = [mockDocuments[0], mockDocuments[1]];
      
      act(() => {
        result.current.selectDocuments(docs);
      });
      
      expect(result.current.selectedDocuments).toHaveLength(2);
    });

    it('应该能够切换文档选择状态', () => {
      const { result } = renderHook(() => useKnowledgeStore());
      const doc = mockDocuments[0];
      
      act(() => {
        result.current.toggleDocumentSelection(doc);
      });
      
      expect(result.current.selectedDocuments).toContainEqual(doc);
      
      act(() => {
        result.current.toggleDocumentSelection(doc);
      });
      
      expect(result.current.selectedDocuments).not.toContainEqual(doc);
    });
  });
});

import { Outlet, useNavigate, useLocation } from 'react-router-dom';
import { useKnowledgeStore } from '../stores/knowledgeStore';
import KnowledgeSidebar from '../components/Knowledge/KnowledgeSidebar';
import KnowledgeToolbar from '../components/Knowledge/KnowledgeToolbar';
import './KnowledgeLayout.css';

const KnowledgeLayout = () => {
  const navigate = useNavigate();
  const location = useLocation();
  const { activeTab, setActiveTab, sidebarCollapsed } = useKnowledgeStore();
  
  // 根据 URL 同步 tab 状态
  useEffect(() => {
    const path = location.pathname.split('/').pop();
    if (path && path !== activeTab) {
      setActiveTab(path);
    }
  }, [location, activeTab, setActiveTab]);
  
  const handleTabChange = (tab) => {
    setActiveTab(tab);
    navigate(`/knowledge/${tab}`);
  };
  
  return (
    <div className={`knowledge-layout ${sidebarCollapsed ? 'sidebar-collapsed' : ''}`}>
      <KnowledgeSidebar />
      <div className="knowledge-main">
        <KnowledgeToolbar activeTab={activeTab} onTabChange={handleTabChange} />
        <div className="knowledge-content">
          <Outlet />
        </div>
      </div>
    </div>
  );
};

// 3. 创建文档管理页面
// src/pages/knowledge/DocumentManagement.jsx

import { useEffect } from 'react';
import { useParams } from 'react-router-dom';
import { useKnowledgeStore } from '../../stores/knowledgeStore';
import { useDocumentQuery } from '../../hooks/queries/useDocumentQuery';
import DocumentList from '../../components/Knowledge/DocumentList';
import DocumentDetail from '../../components/Knowledge/DocumentDetail';
import BatchOperationToolbar from '../../components/Knowledge/BatchOperationToolbar';
import './DocumentManagement.css';

const DocumentManagement = () => {
  const { documentId } = useParams();
  const { 
    currentKnowledgeBase, 
    documents, 
    selectedDocuments,
    viewMode,
    documentFilters 
  } = useKnowledgeStore();
  
  // 使用 React Query 获取文档数据
  const { data, isLoading, error } = useDocumentQuery({
    knowledgeBaseId: currentKnowledgeBase?.id,
    filters: documentFilters
  });
  
  useEffect(() => {
    if (data) {
      useKnowledgeStore.setState({ documents: data.documents });
    }
  }, [data]);
  
  if (!currentKnowledgeBase) {
    return <EmptyState message="请选择一个知识库" />;
  }
  
  return (
    <div className="document-management">
      <BatchOperationToolbar />
      <div className="document-management-content">
        <DocumentList 
          documents={documents}
          loading={isLoading}
          error={error}
          viewMode={viewMode}
          selectedDocuments={selectedDocuments}
        />
        {documentId && <DocumentDetail documentId={documentId} />}
      </div>
    </div>
  );
};

export default DocumentManagement;

// 1. 虚拟列表组件
// src/components/common/VirtualList/VirtualList.tsx

import React, { useRef, useCallback } from 'react';
import { useVirtualizer } from '@tanstack/react-virtual';
import './VirtualList.css';

interface VirtualListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  estimateSize: number;
  overscan?: number;
  onEndReached?: () => void;
  endReachedThreshold?: number;
  className?: string;
}

export function VirtualList<T>({
  items,
  renderItem,
  estimateSize,
  overscan = 5,
  onEndReached,
  endReachedThreshold = 200,
  className
}: VirtualListProps<T>) {
  const parentRef = useRef<HTMLDivElement>(null);
  const endReachedCalledRef = useRef(false);

  const virtualizer = useVirtualizer({
    count: items.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => estimateSize,
    overscan,
    measureElement: (el) => el.getBoundingClientRect().height
  });

  // 监听滚动，触发加载更多
  const handleScroll = useCallback(() => {
    if (!onEndReached || !parentRef.current) return;

    const { scrollTop, scrollHeight, clientHeight } = parentRef.current;
    const scrollBottom = scrollHeight - scrollTop - clientHeight;

    if (scrollBottom < endReachedThreshold && !endReachedCalledRef.current) {
      endReachedCalledRef.current = true;
      onEndReached();
    } else if (scrollBottom >= endReachedThreshold) {
      endReachedCalledRef.current = false;
    }
  }, [onEndReached, endReachedThreshold]);

  const virtualItems = virtualizer.getVirtualItems();

  return (
    <div
      ref={parentRef}
      className={`virtual-list ${className || ''}`}
      onScroll={handleScroll}
    >
      <div
        className="virtual-list__content"
        style={{
          height: `${virtualizer.getTotalSize()}px`,
          width: '100%',
          position: 'relative'
        }}
      >
        {virtualItems.map((virtualItem) => (
          <div
            key={virtualItem.key}
            ref={virtualizer.measureElement}
            data-index={virtualItem.index}
            className="virtual-list__item"
            style={{
              position: 'absolute',
              top: 0,
              left: 0,
              width: '100%',
              transform: `translateY(${virtualItem.start}px)`
            }}
          >
            {renderItem(items[virtualItem.index], virtualItem.index)}
          </div>
        ))}
      </div>
    </div>
  );
}

const documentKeys = {

  });

    getNextPageParam: (lastPage) => {
      if (lastPage.page >= lastPage.totalPages) return undefined;
      return lastPage.page + 1;
    },
    enabled: !!kbId

};

import { DocumentList } from '../components/Knowledge/DocumentList';
import { generateMockDocuments } from '../test/fixtures';

describe('Knowledge Module Performance', () => {
  it('应该在大数据量下保持流畅', async () => {
    const largeDataset = generateMockDocuments(10000);
    
    const startTime = performance.now();
    
    const { container } = render(<DocumentList documents={largeDataset} />);
    
    const renderTime = performance.now() - startTime;
    
    // 渲染时间应该小于 1秒
    expect(renderTime).toBeLessThan(1000);
    
    // 检查虚拟列表是否正确渲染
    const visibleItems = container.querySelectorAll('.document-card');
    expect(visibleItems.length).toBeLessThan(50); // 只渲染可见项
  });

  it('应该快速响应用户交互', async () => {
    const documents = generateMockDocuments(1000);
    
    const { container } = render(<DocumentList documents={documents} />);
    
    const startTime = performance.now();
    
    // 模拟快速滚动
    const list = container.querySelector('.virtual-list');
    list?.scrollTo(0, 5000);
    
    const scrollTime = performance.now() - startTime;
    
    // 滚动响应时间应该小于 16ms (60fps)
    expect(scrollTime).toBeLessThan(16);
  });
});
