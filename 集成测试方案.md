```python
import logging
```

```python
# test_batch_api.py
import requests
import time

BASE_URL = "http://localhost:8000/api/v1"

def test_batch_process():
    """测试批量处理API"""
    
    # 1. 登录获取token
    login_response = requests.post(f"{BASE_URL}/auth/login", json={
        "username": "test_user",
        "password": "test_password"
    })
    token = login_response.json().get("access_token")
    headers = {"Authorization": f"Bearer {token}"}
    
    # 2. 创建知识库
    kb_response = requests.post(f"{BASE_URL}/knowledge-bases", json={
        "name": "批量测试知识库",
        "description": "用于批量处理测试"
    }, headers=headers)
    
    kb_id = kb_response.json().get("id")
    print(f"✅ 创建知识库: {kb_id}")
    
    # 3. 上传多个文档
    document_ids = []
    for i in range(3):
        file_content = f"这是测试文档 {i+1} 的内容。包含实体如苹果公司、史蒂夫·乔布斯等。"
        
        upload_response = requests.post(
            f"{BASE_URL}/knowledge-bases/{kb_id}/documents",
            files={"file": (f"test_{i+1}.txt", file_content.encode())},
            headers=headers
        )
        
        doc_id = upload_response.json().get("id")
        document_ids.append(doc_id)
        print(f"✅ 上传文档 {i+1}: {doc_id}")
    
    # 4. 提交批量处理任务
    batch_response = requests.post(
        f"{BASE_URL}/knowledge-bases/{kb_id}/documents/batch-process",
        json={"document_ids": document_ids},
        headers=headers
    )
    
    task_id = batch_response.json().get("task_id")
    print(f"✅ 批量任务创建: {task_id}")
    
    # 5. 轮询任务状态
    for _ in range(10):
        time.sleep(2)
        
        status_response = requests.get(
            f"{BASE_URL}/knowledge-bases/{kb_id}/documents/batch-status/{task_id}",
            headers=headers
        )
        
        status = status_response.json()
        print(f"   进度: {status['progress']['percentage']}% "
              f"({status['progress']['completed']}/{status['progress']['total']})")
        
        if status['status'] == 'completed':
            print(f"✅ 批量处理完成")
            print(f"   成功: {status['progress']['successful']}")
            print(f"   失败: {status['progress']['failed']}")
            break
    
    return True

if __name__ == "__main__":
    try:
        test_batch_process()
    except Exception as e:
        print(f"❌ 测试失败: {e}")
```

**通过标准：**
- 批量任务创建成功
- 任务状态可查询
- 所有文档处理完成

```python
        assert graph_data, "❌ 知识图谱构建失败"

        print(f"✅ Step 7: 知识图谱 ({len(nodes)} 节点, {len(edges)} 边)")
```

```python
# test_performance.py
import time

def test_single_document_performance():
    """测试单文档处理性能"""
    
    from app.core.database import SessionLocal
    from app.services.knowledge.core.document_processor import DocumentProcessor
    from app.modules.knowledge.models.knowledge_document import KnowledgeBase, KnowledgeDocument
    
    db = SessionLocal()
    processor = DocumentProcessor()
    
    try:
        kb = KnowledgeBase(name="性能测试")
        db.add(kb)
        db.commit()
        
        # 创建不同大小的测试文档
        test_cases = [
            ("小文档", 100),    # 100字符
            ("中文档", 1000),   # 1000字符
            ("大文档", 10000),  # 10000字符
        ]
        
        print("单文档处理性能测试")
        print("="*60)
        
        for name, size in test_cases:
            # 创建测试文档
            content = "这是一个测试句子。" * (size // 10)
            filename = f"perf_test_{size}.txt"
            
            with open(filename, "w", encoding="utf-8") as f:
                f.write(content)
            
            doc = KnowledgeDocument(
                knowledge_base_id=kb.id,
                title=f"性能测试-{name}",
                file_path=filename,
                file_type="txt"
            )
            db.add(doc)
            db.commit()
            
            # 测量处理时间
            start_time = time.time()
            
            result = processor.process_document(
                file_path=filename,
                file_type="txt",
                document_id=doc.id,
                knowledge_base_id=kb.id,
                db=db
            )
            
            elapsed_time = time.time() - start_time
            
            print(f"{name} ({size}字符):")
            print(f"  处理时间: {elapsed_time:.2f}秒")
            print(f"  成功率: {result.get('vectorization_rate', 0):.2%}")
            print(f"  实体数: {len(result.get('entities', []))}")
            print()
        
        print("="*60)
        print("✅ 性能测试完成")
        
    finally:
        db.close()

if __name__ == "__main__":
    test_single_document_performance()
```

**性能指标：**

1. **ImportError: No module named 'xxx'**
   - 检查 `__init__.py` 是否存在
   - 检查导入路径是否正确

2. **实体对齐失败**
   - 检查数据库连接
   - 检查 DocumentEntity 表结构

3. **向量化失败**
   - 检查 ChromaDB 服务
   - 检查 embedding 模型

4. **任务状态查询失败**
   - 检查 Celery 配置
   - 检查 Redis/RabbitMQ 连接
